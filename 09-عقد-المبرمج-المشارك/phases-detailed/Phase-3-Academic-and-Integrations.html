-- 1. Leads (المهتمّون / مرشّحو التسجيل)
CREATE TABLE leads (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  branch_id       UUID REFERENCES branches(id),
  full_name       TEXT NOT NULL,
  phone           TEXT NOT NULL,
  email           TEXT,
  national_id     TEXT,
  birth_date      DATE,
  gender          TEXT CHECK (gender IN ('male', 'female')),
  city            TEXT,
  source          TEXT CHECK (source IN (
    'website', 'whatsapp', 'referral',
    'walk_in', 'ad_campaign', 'event'
  )),
  interested_program_id UUID REFERENCES programs(id),
  stage           TEXT NOT NULL DEFAULT 'new' CHECK (stage IN (
    'new',           -- مهتم جديد
    'contacted',     -- تم التواصل
    'qualified',     -- مؤهَّل
    'visit_scheduled',-- زيارة مجدولة
    'documents_received',-- استلام المستندات
    'payment_pending',-- بانتظار الدفع
    'enrolled'        -- مسجَّل (نهائي)
  )),
  stage_changed_at TIMESTAMPTZ DEFAULT NOW(),
  assigned_to     UUID REFERENCES users(id), -- موظف المبيعات
  notes           TEXT,
  utm_source      TEXT,
  utm_campaign    TEXT,
  consent_marketing BOOLEAN DEFAULT FALSE,
  created_at      TIMESTAMPTZ DEFAULT NOW(),
  updated_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_leads_stage ON leads(stage);
CREATE INDEX idx_leads_branch_stage ON leads(branch_id, stage);
CREATE INDEX idx_leads_assigned ON leads(assigned_to);

-- 2. Lead Activities (سجل التفاعل مع المرشّح)
CREATE TABLE lead_activities (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  lead_id     UUID REFERENCES leads(id) ON DELETE CASCADE,
  user_id     UUID REFERENCES users(id),
  activity    TEXT NOT NULL, -- call, whatsapp, email, visit, note
  outcome     TEXT,
  next_action_at TIMESTAMPTZ,
  details     JSONB,
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- 3. Lead Stage History (تاريخ المراحل لتحليل التحويل)
CREATE TABLE lead_stage_history (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  lead_id     UUID REFERENCES leads(id) ON DELETE CASCADE,
  from_stage  TEXT,
  to_stage    TEXT NOT NULL,
  changed_by  UUID REFERENCES users(id),
  reason      TEXT,
  changed_at  TIMESTAMPTZ DEFAULT NOW()
);

🗂 جداول الجداول الدراسية والقاعات

-- 4. Classrooms (القاعات)
CREATE TABLE classrooms (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  branch_id   UUID REFERENCES branches(id) NOT NULL,
  name        TEXT NOT NULL,
  code        TEXT NOT NULL,           -- مثل A101
  capacity    INTEGER NOT NULL CHECK (capacity > 0),
  type        TEXT CHECK (type IN (
    'lecture', 'lab', 'workshop', 'exam_hall'
  )),
  equipment   TEXT[], -- projector, whiteboard, computers...
  is_active   BOOLEAN DEFAULT TRUE,
  UNIQUE(branch_id, code)
);

-- 5. Class Schedules (الجداول الدراسية)
CREATE TABLE class_schedules (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  branch_id       UUID REFERENCES branches(id) NOT NULL,
  program_id      UUID REFERENCES programs(id),
  course_id       UUID REFERENCES courses(id) NOT NULL,
  teacher_id      UUID REFERENCES users(id) NOT NULL,
  classroom_id    UUID REFERENCES classrooms(id) NOT NULL,
  period          TEXT NOT NULL CHECK (period IN ('morning', 'evening', 'weekend')),
  day_of_week     SMALLINT NOT NULL CHECK (day_of_week BETWEEN 0 AND 6),
  start_time      TIME NOT NULL,
  end_time        TIME NOT NULL CHECK (end_time > start_time),
  semester_id     UUID REFERENCES semesters(id) NOT NULL,
  max_students    INTEGER NOT NULL,
  status          TEXT DEFAULT 'active' CHECK (status IN ('draft', 'active', 'cancelled', 'completed')),
  created_at      TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_schedules_teacher_time ON class_schedules(teacher_id, day_of_week, start_time);
CREATE INDEX idx_schedules_classroom_time ON class_schedules(classroom_id, day_of_week, start_time);

-- 6. Enrollments (تسجيل الطالب في الفصل)
CREATE TABLE enrollments (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  student_id      UUID REFERENCES students(id) NOT NULL,
  schedule_id     UUID REFERENCES class_schedules(id) NOT NULL,
  enrolled_at     TIMESTAMPTZ DEFAULT NOW(),
  status          TEXT DEFAULT 'active' CHECK (status IN (
    'active', 'withdrawn', 'completed', 'banned'
  )),
  final_grade     NUMERIC(5,2),
  UNIQUE(student_id, schedule_id)
);

'use server';

import { z } from 'zod';
import { createServerClient } from '@/lib/supabase/server';
import { requireRole } from '@/lib/auth';
import { audit } from '@/lib/audit';

const TransitionSchema = z.object({
  leadId: z.string().uuid(),
  toStage: z.enum([
    'new', 'contacted', 'qualified', 'visit_scheduled',
    'documents_received', 'payment_pending', 'enrolled'
  ]),
  reason: z.string().optional(),
});

// مصفوفة الانتقالات المسموحة (Whitelist)
const ALLOWED: Record<string, string[]> = {
  new: ['contacted'],
  contacted: ['qualified', 'new'],
  qualified: ['visit_scheduled'],
  visit_scheduled: ['documents_received', 'qualified'],
  documents_received: ['payment_pending', 'visit_scheduled'],
  payment_pending: ['enrolled', 'documents_received'],
  enrolled: [],
};

export async function transitionLead(input: z.infer<typeof TransitionSchema>) {
  const parsed = TransitionSchema.parse(input);
  const user = await requireRole(['admin', 'sales', 'registrar']);
  const supabase = createServerClient();

  const { data: lead, error } = await supabase
    .from('leads')
    .select('id, stage, branch_id')
    .eq('id', parsed.leadId)
    .single();
  if (error || !lead) throw new Error('Lead not found');

  // Guard: الانتقال مسموح؟
  if (!ALLOWED[lead.stage]?.includes(parsed.toStage)) {
    throw new Error(`Transition not allowed: ${lead.stage} → ${parsed.toStage}`);
  }

  // التحديث + التاريخ
  const { error: updateErr } = await supabase.from('leads').update({
    stage: parsed.toStage,
    stage_changed_at: new Date().toISOString(),
    updated_at: new Date().toISOString(),
  }).eq('id', parsed.leadId);
  if (updateErr) throw updateErr;

  await supabase.from('lead_stage_history').insert({
    lead_id: parsed.leadId,
    from_stage: lead.stage,
    to_stage: parsed.toStage,
    changed_by: user.id,
    reason: parsed.reason,
  });

  // إذا وصل لـenrolled → إنشاء طالب فعلي
  if (parsed.toStage === 'enrolled') {
    await convertLeadToStudent(parsed.leadId, user.id);
  }

  await audit({ entity: 'lead', entityId: parsed.leadId, action: 'transition', meta: parsed });
  return { success: true, newStage: parsed.toStage };
}

// src/server/cron/attendance-daily.ts
import { adminClient } from '@/lib/supabase/admin';
import { sendWarning } from '@/lib/notifications';
import { log } from '@/lib/logger';

const THRESHOLDS = {
  warning1: 15, // 15% غياب → إنذار 1
  warning2: 20, // 20% غياب → إنذار 2
  warning3: 23, // 23% غياب → إنذار نهائي
  ban:      25, // 25% غياب → حرمان آلي
};

export async function runDailyAttendanceJob() {
  const supabase = adminClient();
  const startedAt = new Date();
  log.info('attendance.daily.start');

  // 1. تحديث ملخّصات الحضور لكل تسجيل نشط
  const { data: enrollments } = await supabase
    .from('enrollments')
    .select('student_id, schedule_id')
    .eq('status', 'active');

  for (const enr of enrollments ?? []) {
    const summary = await recalculateSummary(enr.student_id, enr.schedule_id);

    // 2. التحقق من العتبات
    const rate = summary.absence_rate;
    let newLevel = summary.warning_level;
    if (rate >= THRESHOLDS.ban) {
      await applyAutoBan(enr.student_id, enr.schedule_id, rate);
      continue;
    }
    if (rate >= THRESHOLDS.warning3 && summary.warning_level < 3) newLevel = 3;
    else if (rate >= THRESHOLDS.warning2 && summary.warning_level < 2) newLevel = 2;
    else if (rate >= THRESHOLDS.warning1 && summary.warning_level < 1) newLevel = 1;

    if (newLevel > summary.warning_level) {
      await issueWarning(enr.student_id, enr.schedule_id, newLevel, rate);
    }
  }

  const durationMs = Date.now() - startedAt.getTime();
  log.info('attendance.daily.end', { durationMs, processed: enrollments?.length });
}

async function applyAutoBan(studentId: string, scheduleId: string, rate: number) {
  const supabase = adminClient();

  // 1. تحديث التسجيل
  await supabase.from('enrollments').update({
    status: 'banned',
  }).match({ student_id: studentId, schedule_id: scheduleId });

  // 2. تسجيل في attendance_summary
  await supabase.from('attendance_summary').update({
    banned_at: new Date().toISOString(),
    warning_level: 3,
  }).match({ student_id: studentId, schedule_id: scheduleId });

  // 3. سجل النشاط للمراجعة
  await supabase.from('audit_log').insert({
    entity: 'enrollment',
    entity_id: `${studentId}:${scheduleId}`,
    action: 'auto_ban_25',
    actor: 'system',
    meta: { absence_rate: rate, threshold: 25 },
  });

  // 4. إشعار الطالب + ولي الأمر + المعلم + الإدارة
  await sendBanNotification(studentId, scheduleId, rate);
}

async function issueWarning(studentId: string, scheduleId: string, level: 1 | 2 | 3, rate: number) {
  const supabase = adminClient();
  const messages = {
    1: 'تنبيه أول: نسبة غيابك تجاوزت 15%. الحدّ المسموح 25%.',
    2: 'إنذار ثانٍ: نسبة غيابك تجاوزت 20%. اقتربت من الحرمان.',
    3: 'إنذار نهائي: نسبة غيابك 23%+. أي غياب إضافي يؤدي للحرمان الآلي.',
  };

  await supabase.from('attendance_warnings').insert({
    student_id: studentId,
    schedule_id: scheduleId,
    level,
    absence_rate: rate,
    message: messages[level],
    delivered_via: ['whatsapp', 'sms', 'in_app'],
  });

  await supabase.from('attendance_summary').update({
    warning_level: level,
  }).match({ student_id: studentId, schedule_id: scheduleId });

  await sendWarning({ studentId, scheduleId, level, rate, message: messages[level] });
}

// src/server/actions/schedules/detect-conflicts.ts
export type ConflictKind =
  | 'teacher_double_booked'
  | 'classroom_double_booked'
  | 'capacity_exceeded'
  | 'student_overlap';

export interface Conflict {
  kind: ConflictKind;
  scheduleId: string;
  conflictsWith: string;
  detail: string;
}

export async function detectConflicts(scheduleId: string): Promise<Conflict[]> {
  const supabase = createServerClient();
  const conflicts: Conflict[] = [];

  const { data: sched } = await supabase
    .from('class_schedules')
    .select('*')
    .eq('id', scheduleId)
    .single();
  if (!sched) throw new Error('Schedule not found');

  // 1. هل المعلم محجوز في نفس الوقت؟
  const { data: teacherOverlap } = await supabase.rpc('find_teacher_overlap', {
    p_teacher_id: sched.teacher_id,
    p_day: sched.day_of_week,
    p_start: sched.start_time,
    p_end: sched.end_time,
    p_semester_id: sched.semester_id,
    p_exclude_id: scheduleId,
  });
  teacherOverlap?.forEach((c: any) =>
    conflicts.push({ kind: 'teacher_double_booked', scheduleId, conflictsWith: c.id, detail: `المعلم محجوز في ${c.course_name}` })
  );

  // 2. القاعة محجوزة؟
  const { data: roomOverlap } = await supabase.rpc('find_classroom_overlap', {
    p_classroom_id: sched.classroom_id,
    p_day: sched.day_of_week,
    p_start: sched.start_time,
    p_end: sched.end_time,
    p_semester_id: sched.semester_id,
    p_exclude_id: scheduleId,
  });
  roomOverlap?.forEach((c: any) =>
    conflicts.push({ kind: 'classroom_double_booked', scheduleId, conflictsWith: c.id, detail: `القاعة محجوزة لـ${c.course_name}` })
  );

  // 3. عدد الطلاب > سعة القاعة؟
  const { count } = await supabase
    .from('enrollments')
    .select('*', { count: 'exact', head: true })
    .eq('schedule_id', scheduleId)
    .eq('status', 'active');
  const { data: room } = await supabase
    .from('classrooms')
    .select('capacity')
    .eq('id', sched.classroom_id)
    .single();
  if ((count ?? 0) > (room?.capacity ?? 0)) {
    conflicts.push({ kind: 'capacity_exceeded', scheduleId, conflictsWith: 'classroom', detail: `${count} طالب > سعة ${room?.capacity}` });
  }

  return conflicts;
}

// src/server/eligibility/engine.ts
export interface EligibilityResult {
  isEligible: boolean;
  attendanceRate: number;
  attendanceOk: boolean;
  assignmentsCompletedPct: number;
  assignmentsOk: boolean;
  financialClear: boolean;
  noActiveSuspension: boolean;
  failureReasons: string[];
}

const RULES = {
  minAttendance: 75,        // 75% حضور
  minAssignmentsCompletion: 80, // 80% من الواجبات مُسلّمة
};

export async function checkEligibility(
  studentId: string,
  examSessionId: string
): Promise<EligibilityResult> {
  const supabase = createServerClient();
  const reasons: string[] = [];

  // 1. الحضور
  const { data: attSummary } = await supabase.rpc('get_avg_attendance_for_student', {
    p_student_id: studentId,
  });
  const attendanceRate = attSummary ?? 0;
  const attendanceOk = attendanceRate >= RULES.minAttendance;
  if (!attendanceOk) reasons.push(`نسبة الحضور ${attendanceRate.toFixed(1)}% < الحد المطلوب ${RULES.minAttendance}%`);

  // 2. الواجبات
  const { data: asnPct } = await supabase.rpc('get_assignments_completion_pct', {
    p_student_id: studentId,
  });
  const assignmentsOk = (asnPct ?? 0) >= RULES.minAssignmentsCompletion;
  if (!assignmentsOk) reasons.push(`نسبة إكمال الواجبات ${asnPct?.toFixed(1)}% < ${RULES.minAssignmentsCompletion}%`);

  // 3. المالي (من البرنامج المحاسبي)
  const balance = await accountingAdapter.getStudentBalance(studentId);
  const financialClear = balance.remainingAmount <= 0 || balance.overdueDays === 0;
  if (!financialClear) reasons.push(`متأخرات مالية ${balance.remainingAmount} ر.س متجاوزة ${balance.overdueDays} يوم`);

  // 4. الإيقاف
  const { data: suspension } = await supabase
    .from('suspensions')
    .select('id')
    .eq('student_id', studentId)
    .or('to_date.gte.now(),is_indefinite.eq.true')
    .limit(1)
    .maybeSingle();
  const noActiveSuspension = !suspension;
  if (!noActiveSuspension) reasons.push('هناك إيقاف نشط على الحساب');

  const isEligible = attendanceOk && assignmentsOk && financialClear && noActiveSuspension;

  // تسجيل الفحص
  await supabase.from('eligibility_checks').upsert({
    student_id: studentId,
    exam_session_id: examSessionId,
    is_eligible: isEligible,
    attendance_rate: attendanceRate,
    attendance_ok: attendanceOk,
    assignments_completed_pct: asnPct ?? 0,
    assignments_ok: assignmentsOk,
    financial_clear: financialClear,
    no_active_suspension: noActiveSuspension,
    failure_reasons: reasons,
  }, { onConflict: 'student_id,exam_session_id' });

  return { isEligible, attendanceRate, attendanceOk, assignmentsCompletedPct: asnPct ?? 0,
    assignmentsOk, financialClear, noActiveSuspension, failureReasons: reasons };
}

// src/server/actions/exams/import-results-excel.ts
'use server';

import * as XLSX from 'xlsx';
import { z } from 'zod';

const RowSchema = z.object({
  student_number: z.string().min(1),
  full_name: z.string().min(1),
  score: z.number().min(0).max(100),
});
type Row = z.infer<typeof RowSchema>;

export interface ImportResult {
  totalRows: number;
  successful: number;
  failed: number;
  errors: { row: number; reason: string }[];
  warnings: { row: number; warning: string }[];
  preview: Array<{ row: number; studentName: string; score: number; status: string }>;
}

export async function parseExcelResults(
  fileBuffer: ArrayBuffer,
  examSessionId: string,
  dryRun: boolean = true
): Promise<ImportResult> {
  await requireRole(['admin', 'academic_supervisor']);

  const wb = XLSX.read(fileBuffer, { type: 'array' });
  const sheet = wb.Sheets[wb.SheetNames[0]];
  const rows = XLSX.utils.sheet_to_json(sheet, { defval: null }) as Record<string, unknown>[];

  const result: ImportResult = {
    totalRows: rows.length, successful: 0, failed: 0,
    errors: [], warnings: [], preview: [],
  };

  const supabase = createServerClient();
  const { data: exam } = await supabase
    .from('final_exam_sessions')
    .select('passing_score, program_id')
    .eq('id', examSessionId)
    .single();
  if (!exam) throw new Error('Exam session not found');

  const toInsert: any[] = [];

  for (let i = 0; i < rows.length; i++) {
    const raw = rows[i];
    const rowNum = i + 2; // Excel row number (with header)
    const parsed = RowSchema.safeParse({
      student_number: String(raw['رقم الطالب'] ?? raw.student_number ?? ''),
      full_name: String(raw['الاسم'] ?? raw.full_name ?? ''),
      score: Number(raw['الدرجة'] ?? raw.score),
    });
    if (!parsed.success) {
      result.errors.push({ row: rowNum, reason: parsed.error.errors.map(e => e.message).join(', ') });
      result.failed++;
      continue;
    }
    const row: Row = parsed.data;

    // تحقق من وجود الطالب وأهليته
    const { data: student } = await supabase
      .from('students')
      .select('id, full_name')
      .eq('student_number', row.student_number)
      .maybeSingle();
    if (!student) {
      result.errors.push({ row: rowNum, reason: `الطالب ${row.student_number} غير موجود` });
      result.failed++;
      continue;
    }

    // تحقق من أن الاسم مطابق (warning فقط)
    if (student.full_name.replace(/\s/g, '') !== row.full_name.replace(/\s/g, '')) {
      result.warnings.push({ row: rowNum, warning: `الاسم لا يطابق: "${student.full_name}" vs "${row.full_name}"` });
    }

    const { data: elig } = await supabase
      .from('eligibility_checks')
      .select('is_eligible')
      .match({ student_id: student.id, exam_session_id: examSessionId })
      .maybeSingle();
    if (!elig?.is_eligible) {
      result.errors.push({ row: rowNum, reason: `${row.full_name} غير مؤهَّل للاختبار` });
      result.failed++;
      continue;
    }

    toInsert.push({
      exam_session_id: examSessionId,
      student_id: student.id,
      score: row.score,
      passed: row.score >= (exam.passing_score ?? 60),
      uploaded_from: 'excel',
    });
    result.preview.push({
      row: rowNum, studentName: student.full_name, score: row.score,
      status: row.score >= (exam.passing_score ?? 60) ? 'ناجح' : 'راسب',
    });
    result.successful++;
  }

  // حفظ فعلي (إذا dryRun=false)
  if (!dryRun && toInsert.length > 0) {
    const { error } = await supabase.from('final_exam_results').upsert(toInsert, {
      onConflict: 'exam_session_id,student_id',
    });
    if (error) throw error;

    // Audit للاستيراد الكلي
    await audit({
      entity: 'final_exam_results', entityId: examSessionId,
      action: 'import_excel',
      meta: { count: toInsert.length, warnings: result.warnings.length },
    });
  }

  return result;
}

import { PDFDocument, StandardFonts, rgb } from 'pdf-lib';
import fontkit from '@pdf-lib/fontkit';
import { readFile } from 'fs/promises';

export async function generateTranscript(studentId: string): Promise<Uint8Array> {
  const data = await fetchStudentAcademicHistory(studentId);

  const pdf = await PDFDocument.create();
  pdf.registerFontkit(fontkit);
  const cairoFont = await pdf.embedFont(await readFile('./public/fonts/Cairo-Regular.ttf'));

  const page = pdf.addPage([595, 842]); // A4
  page.drawText('كشف درجات', { x: 250, y: 800, size: 22, font: cairoFont, color: rgb(0.03, 0.37, 0.27) });
  page.drawText(`الطالب: ${data.fullName}`, { x: 50, y: 750, size: 14, font: cairoFont });
  page.drawText(`الرقم: ${data.studentNumber}`, { x: 50, y: 730, size: 12, font: cairoFont });

  let y = 690;
  data.courses.forEach(c => {
    page.drawText(`${c.name}: ${c.grade}/100 (${c.letter})`, { x: 50, y, size: 11, font: cairoFont });
    y -= 18;
  });

  page.drawText(`المعدل: ${data.gpa.toFixed(2)}`, { x: 50, y: y - 30, size: 14, font: cairoFont });

  // QR للتحقق
  await drawQRCode(page, `https://verify.iims.app/transcript/${data.transcriptToken}`, 450, 50);

  return await pdf.save();
}

  const windowOpen = await isEvaluationWindowOpen(parsed.scheduleId);
  if (!windowOpen) throw new Error('نافذة التقييم مغلقة');

  // 3. اِبنِ hash لمنع التكرار — لا يكشف هوية الطالب لأنه لا يُخزَّن student_id
  const secret = process.env.EVAL_HMAC_SECRET!;
  const hash = createHmac('sha256', secret)
    .update(`${user.id}:${parsed.scheduleId}`)
    .digest('hex');

  // 4. استرجع المعلم من الجدول
  const { data: sched } = await supabase
    .from('class_schedules')
    .select('teacher_id, semester_id')
    .eq('id', parsed.scheduleId)
    .single();

  // 5. أَدخِل التقييم بدون student_id
  const { error } = await supabase.from('teacher_evaluations').insert({
    teacher_id: sched!.teacher_id,
    schedule_id: parsed.scheduleId,
    semester_id: sched!.semester_id,
    submission_hash: hash,
    clarity: parsed.clarity,
    knowledge: parsed.knowledge,
    fairness: parsed.fairness,
    engagement: parsed.engagement,
    overall: parsed.overall,
    comments: parsed.comments,
  });
  if (error?.code === '23505') throw new Error('سبق وقمت بتقييم هذه المادة');
  if (error) throw error;

  return { success: true };
}

// src/server/integrations/tvtc/adapter.ts
import { CircuitBreaker } from '@/lib/resilience/circuit-breaker';
import { retry } from '@/lib/resilience/retry';
import { logger } from '@/lib/logger';

export class TVTCAdapter implements ITVTCService {
  private readonly baseUrl: string;
  private readonly apiKey: string;
  private readonly breaker: CircuitBreaker;

  constructor(config: { baseUrl: string; apiKey: string }) {
    this.baseUrl = config.baseUrl;
    this.apiKey = config.apiKey;
    this.breaker = new CircuitBreaker({
      failureThreshold: 5,
      resetTimeoutMs: 60_000,
      halfOpenMaxCalls: 1,
    });
  }

  async registerTrainee(student: Student): Promise<TVTCTraineeId> {
    return this.breaker.execute(async () => {
      const idempotencyKey = `reg-${student.id}-${student.programId}`;

      const result = await retry(async () => {
        const res = await fetch(`${this.baseUrl}/api/v1/trainees`, {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json',
            'X-Idempotency-Key': idempotencyKey,
            'X-Branch-Code': student.branchCode,
          },
          body: JSON.stringify({
            nationalId: student.nationalId,
            fullNameAr: student.fullName,
            fullNameEn: student.fullNameEn,
            birthDate: student.birthDate.toISOString().slice(0, 10),
            gender: student.gender,
            phone: student.phone,
            email: student.email,
            programCode: student.programCode,
            enrollmentDate: student.enrollmentDate.toISOString().slice(0, 10),
          }),
          signal: AbortSignal.timeout(15_000),
        });
        if (!res.ok) {
          const body = await res.text();
          throw new Error(`TVTC register failed [${res.status}]: ${body}`);
        }
        return res.json();
      }, { attempts: 3, backoff: 'exponential', initialDelayMs: 1000 });

      logger.info('tvtc.trainee.registered', { studentId: student.id, traineeId: result.traineeId });
      return result.traineeId as TVTCTraineeId;
    });
  }

  async submitGrades(traineeId: TVTCTraineeId, grades: Grade[]): Promise<SubmissionResult> {
    return this.breaker.execute(async () => {
      const idempotencyKey = `grades-${traineeId}-${grades[0]?.semester}-${grades.length}`;
      const payload = {
        traineeId,
        grades: grades.map(g => ({
          courseCode: g.courseCode,
          semester: g.semester,
          score: g.score,
          letter: g.letter,
          isPassing: g.isPassing,
          examType: g.examType,
          examDate: g.examDate.toISOString().slice(0, 10),
        })),
      };

      const res = await retry(async () => {
        const r = await fetch(`${this.baseUrl}/api/v1/grades`, {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${this.apiKey}`,
            'Content-Type': 'application/json',
            'X-Idempotency-Key': idempotencyKey,
          },
          body: JSON.stringify(payload),
          signal: AbortSignal.timeout(30_000),
        });
        if (r.status === 422) {
          const body = await r.json();
          return { acknowledgmentId: '', status: 'rejected' as const, message: 'Validation failed',
            errors: body.errors, receivedAt: new Date() };
        }
        if (!r.ok) throw new Error(`TVTC grades [${r.status}]`);
        const body = await r.json();
        return { acknowledgmentId: body.ackId, status: 'received' as const, receivedAt: new Date() };
      }, { attempts: 3, backoff: 'exponential' });

      return res;
    });
  }

  // ... باقي العمليات (submitAttendance, requestCertificate, reconcileGrades, healthCheck)
}

import { tvtcAdapter } from '@/server/integrations/tvtc';

export async function runTVTCSyncWorker() {
  const supabase = adminClient();
  const { data: jobs } = await supabase
    .from('tvtc_outbox')
    .select('*')
    .eq('status', 'pending')
    .lte('next_attempt_at', new Date().toISOString())
    .order('next_attempt_at')
    .limit(20);

  for (const job of jobs ?? []) {
    // Lock العملية (optimistic)
    const { error: lockErr } = await supabase
      .from('tvtc_outbox')
      .update({ status: 'in_progress', attempts: job.attempts + 1 })
      .match({ id: job.id, status: 'pending' });
    if (lockErr) continue;

    const started = Date.now();
    try {
      let result: any;
      switch (job.operation) {
        case 'register_trainee':
          result = await tvtcAdapter.registerTrainee(job.payload.student);
          await supabase.from('tvtc_trainees').upsert({
            student_id: job.entity_id,
            tvtc_trainee_id: result,
            registered_at: new Date().toISOString(),
          });
          break;

        case 'submit_grades':
          result = await tvtcAdapter.submitGrades(job.payload.traineeId, job.payload.grades);
          if (result.status === 'rejected') throw new Error(`Validation: ${JSON.stringify(result.errors)}`);
          break;

        case 'submit_attendance':
          result = await tvtcAdapter.submitAttendance(job.payload.branchId, job.payload.period, job.payload.reports);
          break;

        case 'request_certificate':
          result = await tvtcAdapter.requestCertificate(job.payload.studentId, job.payload.programId);
          break;
      }

      await supabase.from('tvtc_outbox').update({
        status: 'completed', result, completed_at: new Date().toISOString(),
      }).eq('id', job.id);

      await logSync(job.id, job.operation, job.payload, 200, result, Date.now() - started, job.attempts + 1, true);

    } catch (err: any) {
      const shouldRetry = job.attempts + 1 < job.max_attempts;
      const nextAttempt = new Date(Date.now() + Math.pow(2, job.attempts) * 60_000);
      await supabase.from('tvtc_outbox').update({
        status: shouldRetry ? 'pending' : 'dead_letter',
        next_attempt_at: nextAttempt.toISOString(),
        last_error: err.message,
      }).eq('id', job.id);

      await logSync(job.id, job.operation, job.payload, 0, null, Date.now() - started, job.attempts + 1, false, err.message);

      if (!shouldRetry) await notifyAdminsOnDeadLetter(job);
    }
  }
}

// src/server/hr/biometric/service.ts
export async function recordBiometric(input: BiometricInput): Promise<AttendanceRecord> {
  const supabase = adminClient();
  let employeeId: string;
  let verified = false;
  let livenessScore: number | null = null;
  let location: any = null;
  let photoUrl: string | null = null;

  switch (input.source) {
    case 'hardware_device': {
      const { data: enrollment } = await supabase
        .from('biometric_enrollments')
        .select('employee_id')
        .match({ device_id: input.deviceId, device_user_id: input.deviceUserId })
        .single();
      if (!enrollment) throw new Error('Unknown biometric user');
      employeeId = enrollment.employee_id;
      verified = true; // الجهاز موثوق بطبيعته
      break;
    }
    case 'mobile_app': {
      employeeId = input.employeeId;
      // 1. Geofence: داخل نطاق الفرع؟
      const branch = await getEmployeeBranch(employeeId);
      const distMeters = haversine(input.location, branch.location);
      if (distMeters > 200) throw new Error(`خارج نطاق الفرع (${Math.round(distMeters)}م)`);
      // 2. Liveness على الصورة
      photoUrl = await uploadSelfie(employeeId, input.photoBase64);
      livenessScore = await runLivenessCheck(input.photoBase64);
      verified = livenessScore >= 80;
      // 3. Face match مع الصورة المرجعية
      const match = await faceMatch(employeeId, input.photoBase64);
      if (!match.matched || match.confidence < 75) verified = false;
      location = input.location;
      break;
    }
    case 'manual_override': {
      await requireRole(['admin', 'hr_manager']);
      employeeId = input.employeeId;
      verified = true; // لكن مع override_by + reason
      break;
    }
  }

  // تحديد class_session_id إن كان معلماً قيد حصته
  const sessionId = await findActiveClassSessionForTeacher(employeeId, new Date());

  const { data, error } = await supabase.from('employee_attendance').insert({
    employee_id: employeeId,
    user_type: sessionId ? 'teacher' : 'employee',
    type: input.type,
    timestamp: ('timestamp' in input ? input.timestamp : new Date()).toISOString(),
    source: input.source,
    device_id: input.source === 'hardware_device' ? input.deviceId : null,
    class_session_id: sessionId,
    location, photo_url: photoUrl, liveness_score: livenessScore,
    verified,
    overridden_by: input.source === 'manual_override' ? input.approverId : null,
    override_reason: input.source === 'manual_override' ? input.reason : null,
  }).select().single();
  if (error) throw error;

  return {
    id: data.id, userId: employeeId, userType: data.user_type,
    type: data.type, timestamp: new Date(data.timestamp), source: data.source,
    deviceId: data.device_id, classSessionId: data.class_session_id, verified: data.verified,
  };
}

export interface LeaveBalance {
  annual: { total: number; used: number; pending: number; remaining: number };
  sick: { total: number; used: number; pending: number; remaining: number };
  emergency: { total: number; used: number; pending: number; remaining: number };
  casual: { total: number; used: number; pending: number; remaining: number };
}

const DEFAULT_ALLOCATION = {
  annual: 21,    // 21 يوم سنوياً (متوسط عمل لائق)
  sick: 30,      // 30 يوم مرضية (15 بأجر كامل + 15 بنصف أجر)
  emergency: 5,
  casual: 7,
};

export async function calculateLeaveBalance(employeeId: string, year: number): Promise<LeaveBalance> {
  const supabase = createServerClient();

  // تأكد من وجود سجلات للسنة، أنشئها لو لم تكن
  for (const type of ['annual', 'sick', 'emergency', 'casual'] as const) {
    await supabase.from('leave_balances').upsert({
      employee_id: employeeId, year, leave_type: type,
      total_days: DEFAULT_ALLOCATION[type],
    }, { onConflict: 'employee_id,year,leave_type', ignoreDuplicates: true });
  }

  // اقرأ الأرصدة الحالية
  const { data: balances } = await supabase
    .from('leave_balances')
    .select('*')
    .match({ employee_id: employeeId, year });

  // حدّث pending_days بناءً على الطلبات قيد المراجعة
  const { data: pending } = await supabase
    .from('leave_requests')
    .select('leave_type, total_days')
    .eq('employee_id', employeeId)
    .in('status', ['submitted', 'manager_review', 'needs_info']);

  const pendingMap = (pending ?? []).reduce<Record<string, number>>((acc, p) => {
    acc[p.leave_type] = (acc[p.leave_type] ?? 0) + p.total_days;
    return acc;
  }, {});

  const result: LeaveBalance = {} as any;
  (balances ?? []).forEach(b => {
    const pendingDays = pendingMap[b.leave_type] ?? 0;
    (result as any)[b.leave_type] = {
      total: b.total_days, used: b.used_days, pending: pendingDays,
      remaining: b.total_days - b.used_days - pendingDays,
    };
  });
  return result;
}

// src/server/analytics/risk-detection.ts
export interface StudentRiskScore {
  studentId: string;
  totalScore: number;          // 0-100, 100=أعلى خطر
  level: 'low' | 'medium' | 'high' | 'critical';
  factors: {
    attendance: { score: number; rate: number; impact: number };
    grades: { score: number; gpa: number; impact: number };
    assignments: { score: number; pct: number; impact: number };
    financial: { score: number; overdueDays: number; impact: number };
    behavioral: { score: number; complaints: number; warnings: number; impact: number };
  };
  recommendations: string[];
  calculatedAt: Date;
}

const WEIGHTS = {
  attendance: 0.30,    // 30%
  grades: 0.25,        // 25%
  assignments: 0.20,   // 20%
  financial: 0.15,     // 15%
  behavioral: 0.10,    // 10%
};

export async function calculateRiskScore(studentId: string): Promise<StudentRiskScore> {
  const supabase = createServerClient();

  // 1. الحضور
  const { data: attRate } = await supabase.rpc('get_avg_attendance_for_student', { p_student_id: studentId });
  const rate = attRate ?? 100;
  const attendanceScore = rate < 70 ? 100 : rate < 80 ? 70 : rate < 90 ? 40 : 10;

  // 2. الدرجات (GPA)
  const { data: gpa } = await supabase.rpc('get_student_gpa', { p_student_id: studentId });
  const gpaVal = gpa ?? 100;
  const gradesScore = gpaVal < 60 ? 100 : gpaVal < 70 ? 70 : gpaVal < 80 ? 30 : 10;

  // 3. الواجبات
  const { data: asnPct } = await supabase.rpc('get_assignments_completion_pct', { p_student_id: studentId });
  const pct = asnPct ?? 100;
  const assignmentsScore = pct < 50 ? 100 : pct < 70 ? 70 : pct < 85 ? 30 : 10;

  // 4. المالي
  const balance = await accountingAdapter.getStudentBalance(studentId);
  const overdueDays = balance.overdueDays;
  const financialScore = overdueDays > 60 ? 100 : overdueDays > 30 ? 70 : overdueDays > 0 ? 30 : 0;

  // 5. السلوكي
  const { data: behavior } = await supabase.rpc('get_student_behavioral_stats', { p_student_id: studentId });
  const complaints = behavior?.complaints_against ?? 0;
  const warnings = behavior?.warnings ?? 0;
  const behavioralScore = Math.min(100, complaints * 30 + warnings * 20);

  const totalScore =
    attendanceScore * WEIGHTS.attendance +
    gradesScore * WEIGHTS.grades +
    assignmentsScore * WEIGHTS.assignments +
    financialScore * WEIGHTS.financial +
    behavioralScore * WEIGHTS.behavioral;

  const level: StudentRiskScore['level'] =
    totalScore >= 75 ? 'critical' :
    totalScore >= 50 ? 'high' :
    totalScore >= 25 ? 'medium' : 'low';

  const recommendations: string[] = [];
  if (rate < 80) recommendations.push('حدّد جلسة متابعة لمناقشة الغياب');
  if (gpaVal < 70) recommendations.push('برنامج دعم أكاديمي + ساعات إضافية');
  if (pct < 70) recommendations.push('متابعة المعلم لتسليم الواجبات المتأخرة');
  if (overdueDays > 30) recommendations.push('جلسة مع ولي الأمر لمعالجة المتأخرات');
  if (complaints > 1) recommendations.push('مقابلة المرشد الأكاديمي للوضع السلوكي');

  return {
    studentId, totalScore: Math.round(totalScore), level,
    factors: {
      attendance: { score: attendanceScore, rate, impact: attendanceScore * WEIGHTS.attendance },
      grades: { score: gradesScore, gpa: gpaVal, impact: gradesScore * WEIGHTS.grades },
      assignments: { score: assignmentsScore, pct, impact: assignmentsScore * WEIGHTS.assignments },
      financial: { score: financialScore, overdueDays, impact: financialScore * WEIGHTS.financial },
      behavioral: { score: behavioralScore, complaints, warnings, impact: behavioralScore * WEIGHTS.behavioral },
    },
    recommendations,
    calculatedAt: new Date(),
  };
}

export interface WhatsAppTemplate {
  name: string;
  language: 'ar' | 'en';
  variables: Record<string, string>;
}

export class UnifonicAdapter {
  constructor(private config: { appSid: string; senderId: string; baseUrl: string }) {}

  async sendWhatsAppTemplate(to: string, template: WhatsAppTemplate) {
    const normalized = this.normalizePhone(to);
    const body = {
      AppSid: this.config.appSid,
      Recipient: normalized,
      TemplateName: template.name,
      Language: template.language,
      Body: this.fillTemplate(template.name, template.variables),
    };

    const res = await fetch(`${this.config.baseUrl}/whatsapp/messages`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
      signal: AbortSignal.timeout(10_000),
    });
    if (!res.ok) throw new Error(`Unifonic WA failed: ${res.status}`);
    const data = await res.json();
    return { messageId: data.data.MessageID, status: 'sent' };
  }

  async sendSMS(to: string, message: string) {
    const normalized = this.normalizePhone(to);
    const body = {
      AppSid: this.config.appSid,
      SenderID: this.config.senderId,
      Recipient: normalized,
      Body: message,
    };
    const res = await fetch(`${this.config.baseUrl}/sms/messages`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    });
    if (!res.ok) throw new Error(`Unifonic SMS failed: ${res.status}`);
    return { messageId: (await res.json()).data.MessageID, status: 'sent' };
  }

  private normalizePhone(p: string): string {
    const digits = p.replace(/\D/g, '');
    if (digits.startsWith('05')) return '966' + digits.slice(1);
    if (digits.startsWith('5') && digits.length === 9) return '966' + digits;
    return digits;
  }

  private fillTemplate(name: string, vars: Record<string, string>): string {
    const tpl = TEMPLATES[name];
    if (!tpl) throw new Error(`Unknown template: ${name}`);
    return tpl.replace(/{{(\w+)}}/g, (_, key) => vars[key] ?? '');
  }
}

- المتأخرات > 30 يوم → حجب الخدمات الأكاديمية.
`;

export async function chatbotResponse(
  studentId: string,
  conversationId: string,
  newMessage: string,
  history: Array<{ role: 'user' | 'assistant'; content: string }>
) {
  const context = await buildStudentContext(studentId);
  const contextJson = JSON.stringify(context, null, 2);

  const stream = await anthropic.messages.stream({
    model: 'claude-opus-4-7',
    max_tokens: 1024,
    system: [
      { type: 'text', text: SYSTEM_PROMPT_BASE,
        cache_control: { type: 'ephemeral' } }, // prompt caching
      { type: 'text',
        text: `<student_context>\n${contextJson}\n</student_context>` },
    ],
    messages: [...history, { role: 'user', content: newMessage }],
  });

  let full = '';
  stream.on('text', (text) => { full += text; });
  const final = await stream.finalMessage();

  // تسجيل المحادثة
  await saveChatMessage({
    conversationId, role: 'user', content: newMessage,
    inputTokens: final.usage.input_tokens,
  });
  await saveChatMessage({
    conversationId, role: 'assistant', content: full,
    outputTokens: final.usage.output_tokens,
    cacheReadTokens: final.usage.cache_read_input_tokens,
    cacheCreationTokens: final.usage.cache_creation_input_tokens,
  });

  return { content: full, stream };
}

📊 Materialized Views للتقارير

التقارير المعقّدة تُحسب على Materialized Views يُجدَّد كل ساعة/يوم بحسب الاستخدام، بدلاً من حساب الـquery في كل طلب.

-- 1. ملخّص أداء الفرع
CREATE MATERIALIZED VIEW mv_branch_performance AS
SELECT
  b.id AS branch_id, b.name,
  COUNT(DISTINCT s.id) AS total_students,
  COUNT(DISTINCT e.id) FILTER (WHERE e.status = 'active') AS active_enrollments,
  AVG(asm.absence_rate) AS avg_absence_rate,
  COUNT(DISTINCT asm.student_id) FILTER (WHERE asm.absence_rate >= 25) AS banned_students,
  AVG(tk.pass_rate) AS avg_teacher_pass_rate
FROM branches b
LEFT JOIN students s ON s.branch_id = b.id
LEFT JOIN enrollments e ON e.student_id = s.id
LEFT JOIN attendance_summary asm ON asm.student_id = s.id
LEFT JOIN teacher_kpis tk ON tk.teacher_id IN (
  SELECT DISTINCT teacher_id FROM class_schedules WHERE branch_id = b.id
)
GROUP BY b.id, b.name;

CREATE UNIQUE INDEX idx_mv_branch_perf ON mv_branch_performance(branch_id);

-- 2. تقرير Risk للطلاب
CREATE MATERIALIZED VIEW mv_student_risk AS
SELECT
  s.id AS student_id, s.full_name, s.branch_id,
  AVG(asm.absence_rate) AS avg_absence,
  COUNT(c.id) FILTER (WHERE c.status IN ('submitted', 'in_progress')) AS open_complaints,
  COUNT(aw.id) AS active_warnings,
  MAX(aw.level) AS max_warning_level
FROM students s
LEFT JOIN attendance_summary asm ON asm.student_id = s.id
LEFT JOIN complaints c ON c.student_id = s.id
LEFT JOIN attendance_warnings aw ON aw.student_id = s.id
  AND aw.created_at > NOW() - INTERVAL '30 days'
GROUP BY s.id;

-- 3. ملخّص استخدام Chatbot
CREATE MATERIALIZED VIEW mv_chatbot_usage AS
SELECT
  DATE(cm.created_at) AS day,
  COUNT(DISTINCT cc.student_id) AS unique_users,
  COUNT(cm.id) AS total_messages,
  AVG(cm.latency_ms) AS avg_latency_ms,
  SUM(cm.input_tokens) AS total_input_tokens,
  SUM(cm.output_tokens) AS total_output_tokens,
  SUM(cm.cache_read_tokens) AS total_cache_reads
FROM chat_messages cm
JOIN chat_conversations cc ON cc.id = cm.conversation_id
GROUP BY DATE(cm.created_at);

$$);
