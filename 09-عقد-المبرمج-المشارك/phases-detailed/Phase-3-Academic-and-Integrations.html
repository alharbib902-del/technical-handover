// app/Services/TVTC/TVTCService.php
<?php

namespace App\Services\TVTC;

use App\Services\TVTC\Contracts\TVTCServiceInterface;
use App\Services\TVTC\DTOs\{TraineeDTO, SubmissionResultDTO};
use App\Services\TVTC\Support\XMLSigner;
use Illuminate\Http\Client\PendingRequest;
use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;
use Illuminate\Support\Str;

class TVTCService implements TVTCServiceInterface
{
    public function __construct(
        private readonly XMLSigner $signer,
        private readonly string $baseUrl,
        private readonly string $apiKey,
    ) {}

    public function registerTrainee(TraineeDTO $trainee): string
    {
        $xml = $this->buildTraineeXML($trainee);
        $signedXml = $this->signer->sign($xml);
        $idempotencyKey = 'reg-' . $trainee->studentId . '-' . $trainee->programCode;

        $response = $this->client()
            ->withHeaders([
                'Content-Type' => 'application/xml',
                'X-Idempotency-Key' => $idempotencyKey,
                'X-Branch-Code' => $trainee->branchCode,
            ])
            ->withBody($signedXml, 'application/xml')
            ->retry(3, 1000, throw: false)
            ->post('/api/v1/trainees');

        if ($response->failed()) {
            Log::error('tvtc.register.failed', [
                'student_id' => $trainee->studentId,
                'status' => $response->status(),
                'body' => $response->body(),
            ]);
            throw new \RuntimeException("TVTC register failed [{$response->status()}]");
        }

        $data = $response->json();
        Log::info('tvtc.register.success', [
            'student_id' => $trainee->studentId,
            'tvtc_id' => $data['traineeId'],
        ]);

        return $data['traineeId'];
    }

    public function submitGrades(string $tvtcId, array $grades): SubmissionResultDTO
    {
        $xml = $this->buildGradesXML($tvtcId, $grades);
        $signedXml = $this->signer->sign($xml);
        $idempotencyKey = 'grades-' . $tvtcId . '-' . Str::substr(md5(serialize($grades)), 0, 16);

        $response = $this->client()
            ->withHeaders(['X-Idempotency-Key' => $idempotencyKey])
            ->withBody($signedXml, 'application/xml')
            ->timeout(30)
            ->post('/api/v1/grades');

        return SubmissionResultDTO::fromHttpResponse($response);
    }

    public function healthCheck(): array
    {
        $start = microtime(true);
        $ok = $this->client()->timeout(5)->get('/health')->successful();
        return ['ok' => $ok, 'latency_ms' => (microtime(true) - $start) * 1000];
    }

    private function client(): PendingRequest
    {
        return Http::baseUrl($this->baseUrl)
            ->withToken($this->apiKey)
            ->acceptJson();
    }
}

// app/Services/TVTC/Support/XMLSigner.php
<?php

namespace App\Services\TVTC\Support;

use RobRichards\XMLSecLibs\XMLSecurityDSig;
use RobRichards\XMLSecLibs\XMLSecurityKey;

class XMLSigner
{
    public function __construct(
        private readonly string $privateKeyPath,
        private readonly string $certPath,
        private readonly ?string $passphrase = null,
    ) {}

    /**
     * توقيع XML بـ XMLDSig + RSA-SHA256 وفق متطلبات TVTC الفعلية.
     */
    public function sign(string $xml): string
    {
        $doc = new \DOMDocument();
        $doc->loadXML($xml);

        $objDSig = new XMLSecurityDSig();
        $objDSig->setCanonicalMethod(XMLSecurityDSig::EXC_C14N);
        $objDSig->addReference(
            $doc,
            XMLSecurityDSig::SHA256,
            ['http://www.w3.org/2000/09/xmldsig#enveloped-signature']
        );

        $objKey = new XMLSecurityKey(XMLSecurityKey::RSA_SHA256, ['type' => 'private']);
        if ($this->passphrase) {
            $objKey->passphrase = $this->passphrase;
        }
        $objKey->loadKey($this->privateKeyPath, true);

        $objDSig->sign($objKey);
        $objDSig->add509Cert(file_get_contents($this->certPath));
        $objDSig->appendSignature($doc->documentElement);

        return $doc->saveXML();
    }
}

// app/Jobs/TVTC/SubmitTVTCRegistrationJob.php
<?php

namespace App\Jobs\TVTC;

use App\Models\Student;
use App\Models\TVTCSyncLog;
use App\Services\TVTC\Contracts\TVTCServiceInterface;
use App\Services\TVTC\DTOs\TraineeDTO;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Foundation\Bus\Dispatchable;
use Illuminate\Queue\InteractsWithQueue;
use Illuminate\Queue\SerializesModels;
use Illuminate\Support\Facades\Log;
use Throwable;

class SubmitTVTCRegistrationJob implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;

    public int $tries = 5;
    public int $maxExceptions = 3;
    public int $timeout = 60;
    public array $backoff = [60, 180, 600, 1800, 3600]; // exponential

    public function __construct(
        public readonly string $studentId,
    ) {
        $this->onQueue('tvtc');
    }

    public function handle(TVTCServiceInterface $tvtc): void
    {
        $start = microtime(true);
        $student = Student::findOrFail($this->studentId);

        if ($student->tvtc_trainee_id) {
            Log::info('tvtc.register.already_registered', ['student' => $student->id]);
            return;
        }

        try {
            $tvtcId = $tvtc->registerTrainee(TraineeDTO::fromStudent($student));

            $student->update([
                'tvtc_trainee_id' => $tvtcId,
                'tvtc_registered_at' => now(),
            ]);

            TVTCSyncLog::create([
                'operation' => 'register_trainee',
                'entity_type' => 'student',
                'entity_id' => $student->id,
                'success' => true,
                'duration_ms' => (microtime(true) - $start) * 1000,
                'attempt' => $this->attempts(),
            ]);
        } catch (Throwable $e) {
            TVTCSyncLog::create([
                'operation' => 'register_trainee',
                'entity_type' => 'student',
                'entity_id' => $student->id,
                'success' => false,
                'error_message' => $e->getMessage(),
                'attempt' => $this->attempts(),
            ]);
            throw $e;
        }
    }

    public function failed(Throwable $e): void
    {
        // Dead Letter: تم تجاوز كل المحاولات
        Log::critical('tvtc.register.dead_letter', [
            'student_id' => $this->studentId,
            'error' => $e->getMessage(),
        ]);

        // إشعار الأدمن
        Notification::route('mail', config('mail.admin'))
            ->notify(new TVTCSyncFailedNotification($this->studentId, $e));
    }
}

🖐 ZKTecoService — التكامل مع جهاز البصمة

هذا مثال كود حقيقي لاستدعاء جهاز ZKTeco عبر TCP/IP باستخدام مكتبة rats/zkteco. الجهاز يستمع على المنفذ 4370 افتراضياً.

// app/Services/Biometric/ZKTecoService.php
<?php

namespace App\Services\Biometric;

use App\Models\BiometricDevice;
use App\Models\AttendanceLog;
use App\Models\BiometricEnrollment;
use Illuminate\Support\Facades\Log;
use Rats\Zkteco\Lib\ZKTeco;

class ZKTecoService
{
    /**
     * سحب سجلات الحضور من جهاز ZKTeco عبر TCP/IP.
     * يتم استدعاؤها من Job مجدول كل 5 دقائق.
     */
    public function fetchAttendanceLogs(BiometricDevice $device): int
    {
        $zk = new ZKTeco($device->ip_address, $device->port);

        if (! $zk->connect()) {
            Log::warning('biometric.zkteco.connect_failed', [
                'device_id' => $device->id,
                'ip' => $device->ip_address,
            ]);
            throw new \RuntimeException("ZKTeco connect failed for device {$device->device_name}");
        }

        try {
            $zk->disableDevice(); // منع التسجيل أثناء السحب
            $logs = $zk->getAttendance();
            $count = 0;

            foreach ($logs as $log) {
                $enrollment = BiometricEnrollment::where([
                    'device_id' => $device->id,
                    'device_user_id' => (string) $log['id'],
                ])->first();

                if (! $enrollment) {
                    Log::info('biometric.unknown_user', [
                        'device' => $device->device_name,
                        'device_user_id' => $log['id'],
                    ]);
                    continue;
                }

                AttendanceLog::firstOrCreate([
                    'employee_id' => $enrollment->employee_id,
                    'device_id' => $device->id,
                    'timestamp' => $log['timestamp'],
                ], [
                    'direction' => $this->resolveDirection($log['type']),
                    'verification_method' => $enrollment->method,
                    'verified' => true,
                    'raw_payload' => $log,
                ]);
                $count++;
            }

            $device->update(['last_sync_at' => now()]);

            // مسح الذاكرة على الجهاز لمنع تكرار السجلات
            if ($count > 0 && config('biometric.clear_after_sync')) {
                $zk->clearAttendance();
            }

            return $count;
        } finally {
            $zk->enableDevice();
            $zk->disconnect();
        }
    }

    /**
     * تسجيل موظف جديد على جهاز البصمة (Enroll)
     */
    public function enrollEmployee(BiometricDevice $device, string $employeeId, string $fullName): string
    {
        $zk = new ZKTeco($device->ip_address, $device->port);
        $zk->connect();

        $deviceUserId = (string) random_int(1000, 99999);
        $zk->setUser($deviceUserId, $deviceUserId, $fullName, password: '', role: 0);
        $zk->disconnect();

        BiometricEnrollment::create([
            'employee_id' => $employeeId,
            'device_id' => $device->id,
            'device_user_id' => $deviceUserId,
            'method' => 'fingerprint',
            'enrolled_at' => now(),
        ]);

        return $deviceUserId;
    }

    private function resolveDirection(int $type): string
    {
        return match($type) {
            0, 4 => 'in',   // Check-In, Overtime-In
            1, 5 => 'out',  // Check-Out, Overtime-Out
            default => 'in',
        };
    }
}

// app/Http/Controllers/WebAuthnController.php
<?php

namespace App\Http\Controllers;

use App\Models\BiometricEnrollment;
use App\Models\AttendanceLog;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;
use Webauthn\PublicKeyCredentialCreationOptions;
use Webauthn\PublicKeyCredentialRequestOptions;
use Webauthn\Server;

class WebAuthnController extends Controller
{
    public function __construct(private readonly Server $server) {}

    /** بدء عملية التسجيل: يرسل التحدّي للهاتف */
    public function registerOptions(Request $request)
    {
        $user = Auth::user();
        $options = $this->server->generatePublicKeyCredentialCreationOptions(
            user: $user->toWebAuthnUser(),
            attestation: PublicKeyCredentialCreationOptions::ATTESTATION_CONVEYANCE_PREFERENCE_DIRECT,
        );

        cache()->put("webauthn:reg:{$user->id}", $options, 300);
        return response()->json($options);
    }

    /** إكمال التسجيل بعد توقيع الجهاز */
    public function registerVerify(Request $request)
    {
        $user = Auth::user();
        $options = cache()->pull("webauthn:reg:{$user->id}");
        abort_if(! $options, 400, 'انتهت صلاحية الجلسة');

        $credential = $this->server->loadAndCheckAttestationResponse(
            $request->getContent(),
            $options,
            $request,
        );

        BiometricEnrollment::create([
            'employee_id' => $user->employee->id,
            'device_id' => null, // no hardware device for mobile
            'device_user_id' => bin2hex($credential->getPublicKeyCredentialId()),
            'method' => 'webauthn',
            'webauthn_credential' => [
                'public_key' => base64_encode($credential->getCredentialPublicKey()),
                'aaguid' => $credential->getAaguid()->toString(),
                'sign_count' => 0,
            ],
            'enrolled_at' => now(),
        ]);

        return response()->json(['success' => true]);
    }

    /** Check-In/Out عبر بصمة الهاتف + Geofence */
    public function verifyAttendance(Request $request)
    {
        $user = Auth::user();
        $data = $request->validate([
            'direction' => 'required|in:in,out',
            'lat' => 'required|numeric',
            'lng' => 'required|numeric',
            'accuracy' => 'required|numeric',
            'webauthn_response' => 'required',
        ]);

        // 1. Geofence check: داخل 200م من الفرع؟
        $branch = $user->employee->branch;
        $distance = $this->haversine(
            $data['lat'], $data['lng'],
            $branch->latitude, $branch->longitude
        );
        abort_if($distance > 200, 422, "خارج نطاق الفرع ({$distance}م)");

        // 2. WebAuthn verification
        $enrollment = BiometricEnrollment::where('employee_id', $user->employee->id)
            ->where('method', 'webauthn')->firstOrFail();

        $verified = $this->server->loadAndCheckAssertionResponse(
            $data['webauthn_response'],
            $enrollment->webauthn_credential,
            $request,
        );

        AttendanceLog::create([
            'employee_id' => $user->employee->id,
            'timestamp' => now(),
            'direction' => $data['direction'],
            'verification_method' => 'webauthn',
            'location' => ['lat' => $data['lat'], 'lng' => $data['lng'], 'accuracy' => $data['accuracy']],
            'verified' => $verified,
        ]);

        return response()->json(['success' => true, 'direction' => $data['direction']]);
    }

    private function haversine(float $lat1, float $lon1, float $lat2, float $lon2): float
    {
        $R = 6371000; // Earth radius in meters
        $dLat = deg2rad($lat2 - $lat1);
        $dLon = deg2rad($lon2 - $lon1);
        $a = sin($dLat/2)**2 + cos(deg2rad($lat1)) * cos(deg2rad($lat2)) * sin($dLon/2)**2;
        return $R * 2 * asin(sqrt($a));
    }
}

export async function checkInWithBiometric(direction) {
  // 1. Get geolocation
  const position = await new Promise((resolve, reject) =>
    navigator.geolocation.getCurrentPosition(resolve, reject, { enableHighAccuracy: true })
  );

  // 2. Request WebAuthn assertion
  const challenge = await (await fetch('/api/webauthn/options')).json();
  const assertion = await navigator.credentials.get({
    publicKey: {
      challenge: new Uint8Array(challenge.challenge),
      allowCredentials: challenge.allowCredentials,
      userVerification: 'required',  // requires biometric (Touch ID / Face ID)
      timeout: 60000,
    }
  });

      accuracy: position.coords.accuracy,
      webauthn_response: assertion,
    })
  });
  return response.json();
}

class ApproveAndDeductBalance extends Transition
{
    public function __construct(private LeaveRequest $leave) {}

    public function handle(): LeaveRequest
    {
        DB::transaction(function () {
            $balance = LeaveBalance::firstOrCreate([
                'employee_id' => $this->leave->employee_id,
                'leave_type_id' => $this->leave->leave_type_id,
                'year' => now()->year,
            ]);

            if ($balance->total_days - $balance->used_days < $this->leave->total_days) {
                throw new \DomainException('الرصيد المتاح لا يكفي');
            }

            $balance->increment('used_days', $this->leave->total_days);
            $balance->decrement('pending_days', $this->leave->total_days);

            $this->leave->state = new Approved($this->leave);
            $this->leave->decided_at = now();
            $this->leave->save();
        });

        $this->leave->employee->user
            ?->notify(new LeaveApprovedNotification($this->leave));

        return $this->leave;
    }
}

// app/Services/Payroll/PayrollService.php
<?php

namespace App\Services\Payroll;

use App\Models\Employee;
use App\Models\LeaveRequest;
use App\Services\Accounting\AccountingServiceInterface;
use Carbon\Carbon;

class PayrollService
{
    public function __construct(
        private readonly AccountingServiceInterface $accounting,
    ) {}

    public function generateMonthlyPayroll(Carbon $month): array
    {
        $results = ['sent' => 0, 'failed' => 0];

        Employee::where('status', 'active')->with('activeContract')->chunk(50,
            function ($employees) use ($month, &$results) {
                foreach ($employees as $employee) {
                    try {
                        $payroll = $this->calculateForEmployee($employee, $month);
                        $this->accounting->postPayroll($payroll);
                        $results['sent']++;
                    } catch (\Throwable $e) {
                        $results['failed']++;
                    }
                }
            }
        );

        return $results;
    }

    public function calculateForEmployee(Employee $employee, Carbon $month): array
    {
        $contract = $employee->activeContract;
        abort_if(! $contract, 422, 'لا يوجد عقد نشط');

        $gross = $contract->basic_salary
               + $contract->housing_allowance
               + $contract->transport_allowance;

        // خصومات الإجازات بدون أجر
        $unpaidDays = LeaveRequest::where('employee_id', $employee->id)
            ->whereHas('leaveType', fn($q) => $q->where('is_paid', false))
            ->whereYear('from_date', $month->year)
            ->whereMonth('from_date', $month->month)
            ->whereState('state', [Approved::class, Active::class, Completed::class])
            ->sum('total_days');

        $daily = $gross / 30;
        $unpaidDeduction = $unpaidDays * $daily;

        return [
            'employee_id' => $employee->id,
            'employee_external_id' => $employee->employee_number,
            'month' => $month->format('Y-m'),
            'basic_salary' => $contract->basic_salary,
            'housing' => $contract->housing_allowance,
            'transport' => $contract->transport_allowance,
            'unpaid_deduction' => round($unpaidDeduction, 2),
            'net' => round($gross - $unpaidDeduction, 2),
        ];
    }
}

// app/Jobs/Hr/DailyAttendanceSettlementJob.php
class DailyAttendanceSettlementJob implements ShouldQueue
{
    use Dispatchable, Queueable;

    public function handle(): void
    {
        $today = today();

        Employee::where('status', 'active')->chunk(100, function ($employees) use ($today) {
            foreach ($employees as $employee) {
                $logs = AttendanceLog::where('employee_id', $employee->id)
                    ->whereDate('timestamp', $today)
                    ->orderBy('timestamp')->get();

                if ($logs->isEmpty()) continue;

                $firstIn = $logs->firstWhere('direction', 'in');
                $lastOut = $logs->where('direction', 'out')->last();

                if (! $firstIn || ! $lastOut) continue;

                $expectedStart = $today->copy()->setTime(8, 0);
                $late = max(0, $firstIn->timestamp->diffInMinutes($expectedStart, false));

                DailyEmployeeAttendance::updateOrCreate(
                    ['employee_id'=>$employee->id, 'work_date'=>$today],
                    [
                        'first_check_in'=>$firstIn->timestamp,
                        'last_check_out'=>$lastOut->timestamp,
                        'total_minutes'=>$firstIn->timestamp->diffInMinutes($lastOut->timestamp),
                        'late_minutes'=>$late,
                        'status'=>'present',
                    ]
                );
            }
        });
    }
}

// app/Notifications/Channels/UnifonicWhatsAppChannel.php
<?php

namespace App\Notifications\Channels;

use App\Models\MessageLog;
use App\Models\OptOut;
use Illuminate\Notifications\Notification;
use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\RateLimiter;
use Illuminate\Support\Str;

class UnifonicWhatsAppChannel
{
    public function send($notifiable, Notification $notification): ?MessageLog
    {
        if (! method_exists($notification, 'toWhatsApp')) return null;

        $recipient = $notifiable->routeNotificationFor('whatsapp', $notification);
        if (! $recipient) return null;

        $normalized = $this->normalizePhone($recipient);

        // 1. Opt-out check (PDPL)
        if (OptOut::where(['recipient'=>$normalized, 'channel'=>'whatsapp'])->exists()) {
            return $this->logResult($notifiable, $normalized, 'opted_out');
        }

        // 2. Rate limiting (per user)
        $key = "wa:user:{$notifiable->id}";
        if (RateLimiter::tooManyAttempts($key, 10)) {
            return $this->logResult($notifiable, $normalized, 'rate_limited');
        }
        RateLimiter::hit($key, decaySeconds: 60);

        $message = $notification->toWhatsApp($notifiable);
        $log = MessageLog::create([
            'id' => Str::uuid(),
            'notifiable_type' => $notifiable::class,
            'notifiable_id' => $notifiable->id,
            'channel' => 'whatsapp',
            'recipient' => $normalized,
            'template_name' => $message->template,
            'payload' => $message->parameters,
            'status' => 'queued',
        ]);

        try {
            $response = Http::timeout(10)
                ->withToken(config('services.unifonic.token'))
                ->retry(2, 500)
                ->post(config('services.unifonic.base_url') . '/whatsapp/messages', [
                    'AppSid' => config('services.unifonic.app_sid'),
                    'Recipient' => $normalized,
                    'TemplateName' => $message->template,
                    'Language' => $message->language,
                    'BodyParameters' => $message->parameters,
                ]);

            if ($response->successful()) {
                $body = $response->json();
                $log->update([
                    'status' => 'sent',
                    'provider_id' => $body['data']['MessageID'] ?? null,
                    'sent_at' => now(),
                ]);
            } else {
                $log->update([
                    'status' => 'failed',
                    'error_message' => $response->body(),
                ]);
            }
        } catch (\Throwable $e) {
            $log->update(['status' => 'failed', 'error_message' => $e->getMessage()]);
        }

        return $log;
    }

    private function normalizePhone(string $phone): string
    {
        $digits = preg_replace('/\D/', '', $phone);
        if (str_starts_with($digits, '05')) return '966' . substr($digits, 1);
        if (str_starts_with($digits, '5') && strlen($digits) === 9) return '966' . $digits;
        return $digits;
    }

    private function logResult($notifiable, string $recipient, string $status): MessageLog
    {
        return MessageLog::create([
            'notifiable_type' => $notifiable::class,
            'notifiable_id' => $notifiable->id,
            'channel' => 'whatsapp',
            'recipient' => $recipient,
            'status' => $status,
        ]);
    }
}

🔐 OTPService — Redis + Brute Force Protection

// app/Services/Auth/OTPService.php
<?php

namespace App\Services\Auth;

use Illuminate\Support\Facades\Cache;
use Illuminate\Support\Facades\Hash;
use Illuminate\Support\Facades\RateLimiter;
use App\Notifications\OtpNotification;

class OTPService
{
    private const TTL = 300; // 5 minutes
    private const MAX_ATTEMPTS = 5;
    private const LOCK_MINUTES = 15;

    public function generate(string $phone): array
    {
        $key = "otp:gen:{$phone}";

        // Throttle generation: max 3 per 15min
        if (RateLimiter::tooManyAttempts($key, 3)) {
            throw new \DomainException('تم تجاوز الحد المسموح. حاول بعد 15 دقيقة.');
        }
        RateLimiter::hit($key, decaySeconds: 900);

        $otp = str_pad(random_int(0, 999999), 6, '0', STR_PAD_LEFT);
        $cacheKey = "otp:{$phone}";

        Cache::put($cacheKey, [
            'hash' => Hash::make($otp),
            'attempts' => 0,
            'created_at' => now()->toIso8601String(),
        ], static::TTL);

        // Send via SMS
        Notification::route('sms', $phone)
            ->notify(new OtpNotification($otp));

        return ['success' => true, 'expires_in' => static::TTL];
    }

    public function verify(string $phone, string $otp): bool
    {
        $lockKey = "otp:lock:{$phone}";
        if (Cache::has($lockKey)) {
            throw new \DomainException('الحساب مقفل مؤقتاً بسبب محاولات فاشلة متعددة.');
        }

        $cacheKey = "otp:{$phone}";
        $stored = Cache::get($cacheKey);

        if (! $stored) {
            throw new \DomainException('رمز OTP غير صالح أو منتهي.');
        }

        $stored['attempts']++;

        if (! Hash::check($otp, $stored['hash'])) {
            if ($stored['attempts'] >= static::MAX_ATTEMPTS) {
                Cache::put($lockKey, true, now()->addMinutes(static::LOCK_MINUTES));
                Cache::forget($cacheKey);
                throw new \DomainException('تجاوزت الحد. تم القفل لمدة 15 دقيقة.');
            }
            Cache::put($cacheKey, $stored, static::TTL);
            return false;
        }

        Cache::forget($cacheKey);
        return true;
    }
}

// app/Services/Notifications/NotificationRouter.php
<?php

namespace App\Services\Notifications;

use App\Models\NotificationPreference;

class NotificationRouter
{
    /** يقرر أي قناة (أو قنوات) بناءً على تفضيل المستخدم + جدوى القناة */
    public function resolveChannels($notifiable): array
    {
        $pref = $notifiable->notification_preference;
        $channels = ['database']; // always in-app

        return match($pref?->preferred_channel) {
            'whatsapp_first' => array_merge($channels, [
                $pref->whatsapp_enabled ? 'whatsapp' : null,
                $pref->sms_enabled ? 'sms' : null, // fallback
            ]),
            'sms_first' => array_merge($channels, [
                $pref->sms_enabled ? 'sms' : null,
                $pref->email_enabled ? 'resend' : null,
            ]),
            'email_only' => array_merge($channels, ['resend']),
            'in_app_only' => $channels,
            default => array_merge($channels, ['whatsapp', 'sms']),
        } |> array_filter(...);
    }
}

// Usage in Notification:
public function via($notifiable): array
{
    return app(NotificationRouter::class)->resolveChannels($notifiable);
}

🔔 Webhook: Unifonic Delivery Status

// app/Http/Controllers/Webhooks/UnifonicWebhookController.php
<?php

namespace App\Http\Controllers\Webhooks;

use App\Http\Controllers\Controller;
use App\Models\MessageLog;
use App\Models\OptOut;
use Illuminate\Http\Request;

class UnifonicWebhookController extends Controller
{
    public function handle(Request $request)
    {
        // 1. Verify signature
        $signature = $request->header('X-Unifonic-Signature');
        $expected = hash_hmac('sha256', $request->getContent(), config('services.unifonic.webhook_secret'));
        abort_unless(hash_equals($expected, $signature), 403);

        $payload = $request->all();

        // 2. Update message status
        if ($payload['event'] === 'message.delivered') {
            MessageLog::where('provider_id', $payload['MessageID'])
                ->update(['status' => 'delivered', 'delivered_at' => now()]);
        }

        if ($payload['event'] === 'message.read') {
            MessageLog::where('provider_id', $payload['MessageID'])
                ->update(['status' => 'read', 'read_at' => now()]);
        }

        // 3. Handle opt-out keyword
        if ($payload['event'] === 'message.received') {
            $text = trim($payload['Body']);
            if (in_array(mb_strtolower($text), ['stop', 'إيقاف', 'الغاء'])) {
                OptOut::updateOrCreate(
                    ['recipient' => $payload['From'], 'channel' => 'whatsapp'],
                    ['reason' => 'user_request', 'opted_out_at' => now()]
                );
            }
        }

        return response()->json(['ok' => true]);
    }
}

// app/Services/AI/ClaudeAIService.php
<?php

namespace App\Services\AI;

use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

class ClaudeAIService
{
    private const MODEL = 'claude-opus-4-7';
    private const API_VERSION = '2023-06-01';
    private const BASE_URL = 'https://api.anthropic.com/v1/messages';

    /** اقتراح أسئلة امتحانية من محتوى الدورة (المعلم يراجع/يوافق) */
    public function suggestExamQuestions(string $courseContent, int $count = 5): array
    {
        $response = Http::withHeaders([
            'x-api-key' => config('services.claude.key'),
            'anthropic-version' => self::API_VERSION,
        ])
        ->timeout(60)
        ->post(self::BASE_URL, [
            'model' => self::MODEL,
            'max_tokens' => 2000,
            'system' => [
                [
                    'type' => 'text',
                    'text' => 'أنت مساعد لمعلم في معهد تدريب سعودي. اقترح أسئلة امتحانية بالعربية الفصحى المبسّطة. كل سؤال يجب أن يكون: واضحاً، متوسط الصعوبة، له إجابة قاطعة، يقيس فهماً لا حفظاً. أعد النتيجة JSON: [{question, type, options?, correct, difficulty, explanation}].',
                    'cache_control' => ['type' => 'ephemeral'], // prompt caching
                ],
            ],
            'messages' => [
                [
                    'role' => 'user',
                    'content' => "اقترح {$count} أسئلة من المحتوى التالي:\n\n{$courseContent}",
                ],
            ],
        ]);

        if ($response->failed()) {
            Log::error('claude.suggest_questions.failed', ['body' => $response->body()]);
            throw new \RuntimeException('فشل استدعاء Claude API');
        }

        $body = $response->json();
        $text = $body['content'][0]['text'] ?? '';

        return [
            'questions' => $this->parseQuestions($text),
            'usage' => $body['usage'],
        ];
    }

    /** Q&A مع طالب + context كامل عن حالته (تستخدمها Livewire) */
    public function chat(array $messages, array $context): array
    {
        $systemPrompt = $this->buildSystemPrompt($context);

        $response = Http::withHeaders([
            'x-api-key' => config('services.claude.key'),
            'anthropic-version' => self::API_VERSION,
        ])
        ->timeout(60)
        ->post(self::BASE_URL, [
            'model' => self::MODEL,
            'max_tokens' => 1024,
            'system' => [
                ['type' => 'text', 'text' => $this->basePrompt(),
                 'cache_control' => ['type' => 'ephemeral']],
                ['type' => 'text', 'text' => $systemPrompt],
            ],
            'messages' => $messages,
        ]);

        return [
            'text' => $response['content'][0]['text'] ?? '',
            'usage' => $response['usage'],
        ];
    }

    private function basePrompt(): string
    {
        return <<<PROMPT
        أنت المساعد الذكي لـ IIMS — معهد تدريب سعودي. مهمتك:
        - الإجابة بالعربية الفصحى المبسّطة.
        - استخدم المعلومات الشخصية للطالب من <student_context> فقط — لا تخمين.
        - لو المعلومة غير موجودة: "هذه المعلومة غير متوفرة، راجع شؤون الطلاب".
        - لا تكشف معلومات طلاب آخرين أو معلومات إدارية داخلية.
        - في حالات الأزمة (انتحار، عنف): اطلب التواصل مع شؤون الطلاب أو الطوارئ.

        السياسات الأساسية:
        - حدّ الغياب 25% → حرمان آلي.
        - الاختبار الشامل يتطلب: حضور ≥75% + إكمال 80% من الواجبات + لا حجب مالي.
        - المتأخرات > 30 يوم → حجب الخدمات.
        PROMPT;
    }

    private function buildSystemPrompt(array $context): string
    {
        return "<student_context>\n" .
            json_encode($context, JSON_UNESCAPED_UNICODE | JSON_PRETTY_PRINT) .
            "\n</student_context>";
    }

    private function parseQuestions(string $text): array
    {
        // Extract JSON array from Claude's response
        preg_match('/\[[\s\S]*\]/', $text, $matches);
        return json_decode($matches[0] ?? '[]', true) ?? [];
    }
}

🔍 RAG Service — Hybrid Retrieval

// app/Services/AI/RAGService.php
<?php

namespace App\Services\AI;

use Illuminate\Support\Facades\DB;

class RAGService
{
    public function __construct(
        private readonly EmbeddingService $embedder,
    ) {}

    /** البحث عن أقرب 5 documents بـcosine similarity */
    public function retrieve(string $query, int $topK = 5): array
    {
        $queryVector = $this->embedder->embedQuery($query);
        $vectorStr = '[' . implode(',', $queryVector) . ']';

        return DB::select("
            SELECT
                id, source_type, source_id, title, content, metadata,
                1 - (embedding <=> ?::vector) AS similarity
            FROM knowledge_base
            WHERE 1 - (embedding <=> ?::vector) > 0.7
            ORDER BY embedding <=> ?::vector
            LIMIT ?
        ", [$vectorStr, $vectorStr, $vectorStr, $topK]);
    }

    /** يبني context للـ Claude بناءً على query الطالب */
    public function buildRagContext(string $question): string
    {
        $docs = $this->retrieve($question, 5);

        if (empty($docs)) return '';

        $context = "<knowledge_base_excerpts>\n";
        foreach ($docs as $doc) {
            $context .= "---\n[{$doc->source_type}] {$doc->title} (similarity: "
                . round($doc->similarity, 3) . ")\n{$doc->content}\n";
        }
        $context .= "</knowledge_base_excerpts>";

        return $context;
    }
}

    public function mount(): void
    {
        $this->conversation = ChatConversation::firstOrCreate([
            'user_id' => auth()->id(),
            'ended_at' => null,
        ]);
        $this->messages = $this->conversation->messages->toArray();
    }

    public function sendMessage(
        ClaudeAIService $ai,
        RAGService $rag,
        StudentContextBuilder $builder,
    ): void {
        $this->validate(['input' => 'required|string|max:2000']);

        $userMessage = ChatMessage::create([
            'conversation_id' => $this->conversation->id,
            'role' => 'user',
            'content' => $this->input,
        ]);
        $this->messages[] = $userMessage->toArray();

        $context = $builder->build(auth()->user()->student?->id);
        $ragContext = $rag->buildRagContext($this->input);

        $history = ChatMessage::where('conversation_id', $this->conversation->id)
            ->latest()->take(10)->get()->reverse()
            ->map(fn($m) => ['role' => $m->role, 'content' => $m->content])
            ->toArray();

        $this->streaming = true;
        $this->input = '';

        // استدعاء Claude مع context + RAG
        $result = $ai->chat($history, [
            'student' => $context,
            'rag' => $ragContext,
        ]);

        $assistantMessage = ChatMessage::create([
            'conversation_id' => $this->conversation->id,
            'role' => 'assistant',
            'content' => $result['text'],
            'input_tokens' => $result['usage']['input_tokens'],
            'output_tokens' => $result['usage']['output_tokens'],
            'cache_read_tokens' => $result['usage']['cache_read_input_tokens'] ?? 0,
        ]);
        $this->messages[] = $assistantMessage->toArray();
        $this->streaming = false;
    }

    public function render()
    {
        return view('livewire.ai.chat-box');
    }
}

متطلبات PDPL الأساسية

  • البيانات داخل المملكة: استضافة على Saudi VPS فقط، Object Storage محلي للنسخ الاحتياطية.
  • Encryption at rest: PostgreSQL مع pgcrypto للحقول الحساسة + Disk encryption (LUKS).

  • Right to access: المستخدم يستطيع تصدير كل بياناته (JSON/PDF) من بوابته.

  • Audit Log كامل: Spatie Activity Log على كل وصول لبيانات حساسة.
  • Data Minimization: لا نجمع بيانات إلا ما يلزم العملية التعليمية.
  • Consent: الموافقة الصريحة للتسويق + الإشعارات قبل الإرسال.

🗑 Right to Erasure — Implementation

// app/Services/Privacy/DataErasureService.php
<?php

namespace App\Services\Privacy;

use App\Models\Student;
use Illuminate\Support\Facades\DB;
use Spatie\Activitylog\Models\Activity;

class DataErasureService
{
    /**
     * حذف نهائي لبيانات الطالب — لا يمكن التراجع.
     * يحتفظ بـpseudonymized data للسجلات المحاسبية (متطلب قانوني آخر).
     */
    public function eraseStudent(Student $student, string $requestedBy, string $reason): void
    {
        DB::transaction(function () use ($student, $requestedBy, $reason) {
            $pseudonymId = 'ERASED-' . substr(hash('sha256', $student->id), 0, 12);

            // 1. Anonymize identifying fields
            $student->update([
                'full_name' => $pseudonymId,
                'national_id' => null,
                'phone' => null,
                'email' => null,
                'birth_date' => null,
                'erased_at' => now(),
            ]);

            // 2. Delete from secondary tables
            $student->messageLogs()->delete();
            $student->chatMessages()->delete();
            $student->complaints()->delete();

            // 3. Anonymize photos in Spatie Media Library
            $student->clearMediaCollection('photos');
            $student->clearMediaCollection('documents');

            // 4. Audit log
            activity()->useLog('privacy')
                ->causedBy($requestedBy)
                ->withProperties([
                    'pseudonym_id' => $pseudonymId,
                    'reason' => $reason,
                ])
                ->log('student_data_erased');
        });
    }
}

🛡 قائمة الامتثال الكاملة لـ PDPL السعودي

هذه القائمة الإلزامية يجب أن تكون 100% خضراء قبل إطلاق IIMS في الإنتاج. كل بند مُختبَر وموثَّق في Runbook.

1. توطين البيانات (Data Localization)

  • VPS مستضاف داخل المملكة (Hostinger KSA / Contabo KSA / Salam Cloud).
  • PostgreSQL backups إلى Object Storage محلي (مثل Salam Cloud S3).
  • كل CDN nodes داخل المملكة (Bunny CDN Saudi region أو STC CDN).
  • لا rsync أو نسخ لبيانات إنتاجية خارج المملكة.
  • API calls لـ Claude/Voyage تُرسل بيانات أقل (anonymized context only).

2. التشفير (Encryption)

  • TLS 1.3 على كل HTTP traffic + HSTS preload.
  • Disk encryption (LUKS) على VPS storage.
  • PostgreSQL pgcrypto على الحقول الحساسة (national_id, iqama_number).
  • Spatie Media Library: encryption-at-rest على المستندات الحساسة.
  • API keys مُشفَّرة في config() + لا تُسجَّل في logs.
  • Sessions مشفَّرة + cookies بـhttpOnly + secure + SameSite=Strict.

3. حقوق صاحب البيانات (Data Subject Rights)

  • Right to Access: php artisan user:export <email> ينشئ JSON/PDF كامل.
  • Right to Rectification: المستخدم يعدّل بياناته من بوابته.
  • Right to Erasure: Filament Action مع تأكيد + reason + Audit.
  • Right to Object: opt-out workflow على كل قناة إشعار.
  • Right to Portability: تصدير JSON قابل للقراءة الآلية.
  • SLA الاستجابة لطلبات الحقوق: ≤ 30 يوم (متطلب PDPL).

4. الموافقة (Consent)

  • موافقة صريحة عند التسجيل (checkbox غير مُفعَّل افتراضياً).
  • موافقة منفصلة لـ: استخدام البيانات الأكاديمية / التسويق / المشاركة مع TVTC.
  • timestamp + IP + user_agent مُسجَّلون مع كل موافقة.
  • سهولة سحب الموافقة من بوابة المستخدم.

5. سجل المعالجة (Records of Processing)

  • Spatie Activity Log فعّال على كل Model يحوي PII.
  • جدول data_processing_register يوثّق الغرض من كل عملية معالجة.
  • تقرير شهري تلقائي للـDPO (Data Protection Officer).
  • كل وصول لبيانات حساسة من admin مُسجَّل في activity log.

6. الأمان (Security Measures)

  • 2FA إلزامي للأدوار الإدارية (admin, hr_manager, finance).
  • Password policy: 10+ characters + complexity + breach check (HIBP).
  • Rate limiting: 30 req/min/user + 100 req/min/IP.
  • OWASP Top 10: لا ثغرات (composer audit + ZAP scan).
  • File upload validation: نوع + حجم + virus scan.
  • SQL Injection: Eloquent only + parameterized queries.
  • XSS: Blade escape automatic + DOMPurify على HTML rendering.

7. الإخطار بالخروقات (Breach Notification)

  • SLA الإخطار: 72 ساعة من اكتشاف الخرق (متطلب PDPL).
  • Runbook خاص للـbreach response (في docs/runbook/breach-response.md).
  • Sentry alerts على anomalies (unusual data access, mass downloads).
  • Contact templates جاهزة للمستخدمين المتأثرين + الجهة المختصة.

8. التدريب والتوثيق

  • كل employee إداري دُرِّب على PDPL + تواقيع NDA.
  • Policies موثّقة: Data Retention, Access Control, Incident Response.
  • Annual review للممارسات + audit خارجي مرة سنوياً.

⚠️ تذكير

PDPL ليس "نُجاهز عند الإطلاق ثم ننسى" — هو التزام مستمر. يجب مراجعة هذه القائمة كل ربع سنة، وعند أي تغيير جوهري في الـSchema أو التكاملات أو الـVPS hosting provider.

📐 Migration Best Practices

💡 قواعد كتابة Migrations في Laravel

  • كل Migration له down() مُختبَر وكامل (يدعم rollback).
  • استخدم foreignUuid() مع constrained() + restrictOnDelete() أو cascadeOnDelete().
  • للحالات (state machines): string('state')->default('initial') مع Spatie Model States.
  • JSONB بدلاً من JSON في PostgreSQL (أسرع query + GIN index).
  • Generated columns عبر storedAs() أو raw DB::statement.
  • كل Migration على branch منفصل + PR review من Codex.
