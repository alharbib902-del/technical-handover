3 · لا يُكتَب سطر كود قبل اعتماد Codex

Codex ليس مُراجِعاً لطيفاً. هو مستشار 100/100 صارم. مهمّته أن يحمي المشروع من قرارات هندسية متسرّعة. هذا البروتوكول مُلزِم في بداية كل أسبوع من المرحلة 2.

3.1 الخطوات السبع لاعتماد أسبوع

بروتوكول الاعتماد المُلزِم لكل أسبوع

  1. تحضير خطة الأسبوع — يُسلّم المبرمج المشارك ملفاً واحداً بصيغة Markdown (docs/phase-2/week-N-plan.md) يضم: الأهداف، المهام اليومية، Migrations، توقيع Controllers/Livewire Components، Filament Resources، سيناريوهات Pest، وقائمة المخاطر.
  2. إرسال للـCodex عبر prompt قياسي: "راجع خطة المرحلة 2 الأسبوع N. قيّم على 100. ارفض إن وجدت أي ثغرة."
  3. تقييم Codex من 100 — يُعطي درجة + قائمة ملاحظات نقطية. درجة 92 = لا يكفي. مطلوب 100/100.
  4. التعديل والإعادة — يتم تعديل الخطة معالجة كل ملاحظة، ثم إعادة العرض. عادةً 2-3 جولات قبل 100/100.
  5. الإذن بالبدء — عند 100/100 يصبح المبرمج مأذوناً ببدء التطوير. يُنشَأ branch جديد: feat/phase-2/week-N.
  6. مراجعة يومية — في نهاية كل يوم عمل، يُرسل diff للـCodex مع وصف ما تمّ. أي ملاحظة Critical تُعالَج في اليوم التالي قبل أي شيء جديد.
  7. اعتماد الإنهاء — في نهاية الأسبوع، Codex يُقيّم النتيجة النهائية على 100. لا يُفتَح PR للـmain قبل 100/100 على المُسلَّم.

// app/Services/Accounting/Adapters/OnyxAccountingAdapter.php
declare(strict_types=1);

namespace App\Services\Accounting\Adapters;

use App\Services\Accounting\AccountingAdapterInterface;
use App\Services\Accounting\Exceptions\AccountingException;
use App\DTO\Accounting\InvoiceDTO;
use App\DTO\Accounting\PaymentDTO;
use App\DTO\Accounting\StudentFinancialSnapshot;
use Illuminate\Http\Client\PendingRequest;
use Illuminate\Support\Collection;
use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

final class OnyxAccountingAdapter implements AccountingAdapterInterface
{
    public function __construct(
        private readonly string $baseUrl,
        private readonly string $apiKey,
        private readonly int $timeout = 10,
    ) {}

    public function getStudentFinancialSnapshot(string $externalId): StudentFinancialSnapshot
    {
        $response = $this->client()
            ->get("/api/v2/students/{$externalId}/financial");

        if ($response->failed()) {
            throw new AccountingException(
                "Onyx returned {$response->status()}: {$response->body()}"
            );
        }

        $data = $response->json();

        return new StudentFinancialSnapshot(
            externalId: $externalId,
            totalFees: (float) $data['total_fees'],
            totalPaid: (float) $data['total_paid'],
            overdueAmount: (float) $data['overdue_amount'],
            overdueDays: (int) $data['overdue_days'],
            lastPaymentDate: $data['last_payment_date'] ?? null,
            nextDueDate: $data['next_due_date'] ?? null,
            installmentStatus: $data['status'] ?? 'on_track',
            rawResponse: $data,
        );
    }

    public function getBulkSnapshots(array $externalIds): Collection
    {
        $response = $this->client()
            ->post('/api/v2/students/financial/bulk', [
                'student_ids' => $externalIds,
            ]);

        $response->throw();

        return collect($response->json('students'))
            ->map(fn (array $row) => StudentFinancialSnapshot::fromArray($row));
    }

    public function createInvoice(InvoiceDTO $invoice): string
    {
        $response = $this->client()
            ->post('/api/v2/invoices', $invoice->toArray());

        if ($response->failed()) {
            Log::error('Onyx createInvoice failed', [
                'status' => $response->status(),
                'body'   => $response->body(),
                'invoice'=> $invoice->toArray(),
            ]);
            throw new AccountingException($response->body());
        }

        return $response->json('invoice_id');
    }

    public function recordPayment(PaymentDTO $payment): string
    {
        $response = $this->client()
            ->post('/api/v2/payments', $payment->toArray())
            ->throw();

        return $response->json('payment_id');
    }

    public function healthCheck(): bool
    {
        try {
            return $this->client()
                ->timeout(3)
                ->get('/api/v2/health')
                ->successful();
        } catch (\Throwable $e) {
            return false;
        }
    }

    private function client(): PendingRequest
    {
        return Http::withToken($this->apiKey)
            ->baseUrl($this->baseUrl)
            ->timeout($this->timeout)
            ->retry(3, 200, throw: false)
            ->acceptJson()
            ->asJson();
    }
}

4.9 Webhook Controller — استقبال تحديثات الفواتير

// app/Http/Controllers/Webhooks/AccountingWebhookController.php
declare(strict_types=1);

namespace App\Http\Controllers\Webhooks;

use App\Http\Controllers\Controller;
use App\Jobs\Accounting\ProcessAccountingWebhookJob;
use App\Models\WebhookInbox;
use Illuminate\Http\Request;
use Illuminate\Http\JsonResponse;

final class AccountingWebhookController extends Controller
{
    public function __invoke(Request $request): JsonResponse
    {
        // 1. التحقق من توقيع HMAC SHA-256
        $signature = $request->header('X-Signature', '');
        $secret    = config('accounting.onyx.webhook_secret');
        $expected  = hash_hmac('sha256', $request->getContent(), $secret);

        if (! hash_equals($expected, $signature)) {
            return response()->json(['error' => 'Invalid signature'], 401);
        }

        // 2. Idempotency check
        $key = $request->header('X-Idempotency-Key');
        if (empty($key)) {
            return response()->json(['error' => 'Missing idempotency key'], 400);
        }

        $inbox = WebhookInbox::firstOrCreate(
            ['idempotency_key' => $key],
            [
                'provider'   => 'onyx',
                'event_type' => $request->input('event'),
                'payload'    => $request->all(),
            ]
        );

        if ($inbox->processed_at !== null) {
            // تم استلامه ومعالجته سابقاً — أعد 200 ليتوقف المُرسِل
            return response()->json(['status' => 'already_processed'], 200);
        }

        // 3. Dispatch إلى Queue (لتفادي الـ Timeout)
        ProcessAccountingWebhookJob::dispatch($inbox->idempotency_key)
            ->onQueue('accounting-webhooks');

        return response()->json(['status' => 'queued'], 202);
    }
}

// app/Jobs/Accounting/ProcessAccountingWebhookJob.php
declare(strict_types=1);

namespace App\Jobs\Accounting;

use App\Models\WebhookInbox;
use App\Models\AccountingSnapshot;
use App\Services\Accounting\AccountingAdapterInterface;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Foundation\Bus\Dispatchable;
use Illuminate\Queue\InteractsWithQueue;
use Illuminate\Queue\SerializesModels;
use App\Models\Student;
use App\Services\Blocking\ServiceBlockingEngine;

final class ProcessAccountingWebhookJob implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;

    public int $tries = 5;
    public array $backoff = [10, 30, 60, 300, 900]; // exponential

    public function __construct(public readonly string $idempotencyKey) {}

    public function handle(
        AccountingAdapterInterface $accounting,
        ServiceBlockingEngine $blocking
    ): void
    {
        $inbox = WebhookInbox::findOrFail($this->idempotencyKey);

        // إذا تمت المعالجة بالفعل — اخرج بهدوء (لو re-dispatch)
        if ($inbox->processed_at !== null) {
            return;
        }

        $externalId = data_get($inbox->payload, 'student.external_id');
        $student    = Student::where('external_id', $externalId)->firstOrFail();
        $snapshot   = $accounting->getStudentFinancialSnapshot($externalId);

        AccountingSnapshot::updateOrCreate(
            ['student_id' => $student->id],
            [
                'external_id'        => $snapshot->externalId,
                'total_fees'         => $snapshot->totalFees,
                'total_paid'         => $snapshot->totalPaid,
                'overdue_amount'     => $snapshot->overdueAmount,
                'overdue_days'       => $snapshot->overdueDays,
                'last_payment_date'  => $snapshot->lastPaymentDate,
                'next_due_date'      => $snapshot->nextDueDate,
                'installment_status' => $snapshot->installmentStatus,
                'raw_response'       => $snapshot->rawResponse,
                'last_synced_at'     => now(),
                'source'             => 'webhook',
            ]
        );

        // إعادة تقييم الحجب فوراً بناءً على الحالة المالية الجديدة
        $blocking->reevaluate($student);

        $inbox->update(['processed_at' => now()]);
    }

    public function failed(\Throwable $e): void
    {
        \App\Models\IntegrationError::create([
            'provider'      => 'onyx',
            'endpoint'      => 'webhook',
            'error_message' => $e->getMessage(),
            'severity'      => 'critical',
            'context'       => ['idempotency_key' => $this->idempotencyKey],
        ]);
    }
}

// app/Services/Accounting/CircuitBreaker.php
declare(strict_types=1);

namespace App\Services\Accounting;

use Illuminate\Support\Facades\Cache;

final class CircuitBreaker
{
    private const CACHE_KEY = 'cb:accounting';
    private const THRESHOLD = 5;     // 5 فشل
    private const COOLDOWN = 60;     // 60 ثانية

    public function isOpen(): bool
    {
        $state = Cache::get(self::CACHE_KEY, ['failures' => 0, 'opened_at' => null]);

        if ($state['opened_at'] !== null) {
            $elapsed = now()->diffInSeconds($state['opened_at']);
            if ($elapsed < self::COOLDOWN) {
                return true;
            }
            // نصف-مفتوح: نسمح بمحاولة واحدة
            $this->halfOpen();
        }

        return false;
    }

    public function recordSuccess(): void
    {
        Cache::put(self::CACHE_KEY, ['failures' => 0, 'opened_at' => null], 3600);
    }

    public function recordFailure(): void
    {
        $state = Cache::get(self::CACHE_KEY, ['failures' => 0, 'opened_at' => null]);
        $state['failures']++;

        if ($state['failures'] >= self::THRESHOLD) {
            $state['opened_at'] = now();
        }

        Cache::put(self::CACHE_KEY, $state, 3600);
    }

    private function halfOpen(): void
    {
        Cache::put(self::CACHE_KEY, ['failures' => 4, 'opened_at' => null], 3600);
    }
}

5.6 ServiceBlockingEngine — قلب المحرّك

// app/Services/Blocking/ServiceBlockingEngine.php
declare(strict_types=1);

namespace App\Services\Blocking;

use App\Enums\FinancialStatus;
use App\Enums\InstituteService;
use App\Models\Student;
use App\Models\BlockingRule;
use App\Models\ServiceBlock;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Facades\Cache;

final class ServiceBlockingEngine
{
    private const CACHE_TTL = 600; // 10 دقائق

    /**
     * إعادة تقييم كامل لجميع خدمات الطالب بناءً على حالته المالية.
     * تُستدعى من Observer ومن Job المزامنة.
     */
    public function reevaluate(Student $student): void
    {
        $status = $this->deriveFinancialStatus($student);
        $rules  = $this->rulesForStatus($status);

        DB::transaction(function () use ($student, $rules) {
            foreach (InstituteService::cases() as $service) {
                $rule   = $rules->firstWhere('service', $service->value);
                $active = ServiceBlock::where([
                    'student_id' => $student->id,
                    'service'    => $service->value,
                    'state'      => 'active',
                ])->first();

                if ($rule?->decision === 'block') {
                    $this->applyBlock($student, $service, $rule, $active);
                } elseif ($active) {
                    // القاعدة لم تعد تحجب — ارفع الحجب تلقائياً
                    $active->update([
                        'state'        => 'released',
                        'released_at'  => now(),
                    ]);
                }
            }
        });

        Cache::tags(["student:{$student->id}"])->flush();
    }

    public function canAccess(Student $student, InstituteService $service): BlockingDecision
    {
        return Cache::tags(["student:{$student->id}"])
            ->remember(
                "block:{$student->id}:{$service->value}",
                self::CACHE_TTL,
                fn() => $this->computeDecision($student, $service)
            );
    }

    private function computeDecision(Student $student, InstituteService $service): BlockingDecision
    {
        $block = ServiceBlock::where([
            'student_id' => $student->id,
            'service'    => $service->value,
            'state'      => 'active',
        ])->with('rule')->first();

        if (! $block) {
            return new BlockingDecision(allowed: true);
        }

        return new BlockingDecision(
            allowed: false,
            reason: $block->reason,
            ruleId: $block->rule_id,
            overrideRole: $block->rule->override_role,
        );
    }

    private function deriveFinancialStatus(Student $student): FinancialStatus
    {
        $snap = $student->accountingSnapshot;
        if (! $snap) return FinancialStatus::ON_TRACK;

        return match(true) {
            $student->state->equals('dropped')         => FinancialStatus::DROPPED,
            $snap->overdue_days >= 61               => FinancialStatus::CRITICAL_61_PLUS,
            $snap->overdue_days >= 31               => FinancialStatus::LATE_31_60,
            $snap->overdue_days >= 15               => FinancialStatus::LATE_15_30,
            $snap->overdue_days >= 1                => FinancialStatus::LATE_1_14,
            $snap->total_remaining > 0
                && $snap->total_paid > 0                => FinancialStatus::PARTIAL_PAYMENT,
            $snap->total_remaining <= 0              => FinancialStatus::CLEARED,
            default                                  => FinancialStatus::ON_TRACK,
        };
    }

    private function rulesForStatus(FinancialStatus $status)
    {
        return Cache::tags(['blocking_rules'])
            ->remember("rules:{$status->value}", 3600, fn() =>
                BlockingRule::where('financial_status', $status->value)
                    ->where('is_active', true)
                    ->get()
            );
    }

    private function applyBlock(
        Student $student,
        InstituteService $service,
        BlockingRule $rule,
        ?ServiceBlock $existing
    ): void
    {
        if ($existing && $existing->rule_id === $rule->id) return;

        if ($existing) {
            $existing->update(['state' => 'released', 'released_at' => now()]);
        }

        ServiceBlock::create([
            'student_id'   => $student->id,
            'service'      => $service->value,
            'state'        => 'active',
            'rule_id'      => $rule->id,
            'reason'       => $rule->reason_template,
            'activated_at' => now(),
        ]);
    }
}

// app/Http/Middleware/EnsureServiceUnblocked.php

namespace App\Http\Middleware;

use App\Enums\InstituteService;
use App\Services\Blocking\ServiceBlockingEngine;
use Closure;
use Illuminate\Http\Request;
use Symfony\Component\HttpFoundation\Response;

final class EnsureServiceUnblocked
{
    public function __construct(private ServiceBlockingEngine $engine) {}

    public function handle(Request $request, Closure $next, string $service): Response
    {
        $student = $request->user()?->student;
        if (! $student) {
            return redirect()->route('login');
        }

        $svc      = InstituteService::from($service);
        $decision = $this->engine->canAccess($student, $svc);

        if ($decision->denied()) {
            if ($request->expectsJson()) {
                return response()->json([
                    'error'         => 'service_blocked',
                    'message'       => $decision->reason,
                    'override_role' => $decision->overrideRole,
                ], 403);
            }
            return redirect()->route('student.blocked', [
                'service' => $svc->value,
            ])->with('reason', $decision->reason);
        }

        return $next($request);
    }
}

6.7 WorkflowService — تنسيق الـSubmission والاعتماد

// app/Services/Workflow/WorkflowService.php
declare(strict_types=1);

namespace App\Services\Workflow;

use App\Enums\RequestType;
use App\Models\ApprovalRoute;
use App\Models\RequestEvent;
use App\Models\Student;
use App\Models\StudentRequest;
use App\Models\States\Request\{UnderReview, Approved, Rejected, Processing, Completed, Closed};
use App\Models\User;
use App\Notifications\RequestSubmittedNotification;
use App\Notifications\RequestAssignedNotification;
use App\Notifications\RequestStateChangedNotification;
use Illuminate\Support\Facades\DB;
use Illuminate\Support\Str;

final class WorkflowService
{
    public function submit(
        Student $student,
        RequestType $type,
        string $reason,
        array $data = []
    ): StudentRequest
    {
        return DB::transaction(function () use ($student, $type, $reason, $data) {
            $request = StudentRequest::create([
                'reference'    => $this->generateReference(),
                'student_id'   => $student->id,
                'type'         => $type->value,
                'reason'       => $reason,
                'data'         => $data,
                'sla_days'     => $type->slaDays(),
                'sla_due_at'   => now()->addBusinessDays($type->slaDays()),
                'submitted_at' => now(),
            ]);

            $this->recordEvent($request, 'submitted', comment: 'تقديم الطلب من الطالب');
            $this->assignNextReviewer($request);

            $student->notify(new RequestSubmittedNotification($request));

            return $request;
        });
    }

    public function approve(StudentRequest $request, User $reviewer, ?string $comment = null): void
    {
        $nextRoute = ApprovalRoute::where('request_type', $request->type->value)
            ->where('level', '>', $request->approval_level)
            ->orderBy('level')
            ->first();

        DB::transaction(function () use ($request, $reviewer, $comment, $nextRoute) {
            if ($nextRoute) {
                // انتقل لمستوى تالٍ في سلسلة الاعتماد
                $next = $this->findReviewer($request, $nextRoute->reviewer_role);
                $request->update([
                    'current_reviewer_id' => $next->id,
                    'approval_level'      => $nextRoute->level,
                ]);
                $this->recordEvent($request, 'assigned',
                    actor: $reviewer,
                    comment: $comment ?? 'تمت الموافقة في هذا المستوى وأُحيل للأعلى');
                $next->notify(new RequestAssignedNotification($request));
            } else {
                // اعتماد نهائي → Approved → Processing
                $request->state->transitionTo(Approved::class);
                $this->recordEvent($request, 'state_changed',
                    actor: $reviewer, to: 'approved', comment: $comment);

                $request->state->transitionTo(Processing::class);
                $request->update(['current_reviewer_id' => null]);

                $request->student->notify(new RequestStateChangedNotification($request));
            }
        });
    }

    public function reject(StudentRequest $request, User $reviewer, string $reason): void
    {
        DB::transaction(function () use ($request, $reviewer, $reason) {
            $request->state->transitionTo(Rejected::class);
            $request->update(['completed_at' => now()]);

            $this->recordEvent($request, 'state_changed',
                actor: $reviewer, to: 'rejected', comment: $reason);

            $request->student->notify(new RequestStateChangedNotification($request, $reason));
        });
    }

    public function complete(StudentRequest $request, User $reviewer): void
    {
        $request->state->transitionTo(Completed::class);
        $request->update(['completed_at' => now()]);
        $this->recordEvent($request, 'state_changed',
            actor: $reviewer, to: 'completed');

        $request->student->notify(new RequestStateChangedNotification($request));
    }

    private function assignNextReviewer(StudentRequest $request): void
    {
        $route = ApprovalRoute::where('request_type', $request->type->value)
            ->where('level', 1)
            ->firstOrFail();

        $reviewer = $this->findReviewer($request, $route->reviewer_role);

        $request->update([
            'current_reviewer_id' => $reviewer->id,
            'approval_level'      => 1,
        ]);
        $request->state->transitionTo(UnderReview::class);

        $this->recordEvent($request, 'assigned', comment: "أُحيل إلى {$reviewer->name}");
        $reviewer->notify(new RequestAssignedNotification($request));
    }

    private function findReviewer(StudentRequest $request, string $role): User
    {
        // منطق التوجيه: نفس فرع الطالب، نفس الدور، الأقل عبئاً
        return User::role($role)
            ->where('branch_id', $request->student->branch_id)
            ->withCount(['reviewingRequests' => fn($q) =>
                $q->whereNotIn('state', ['completed', 'closed', 'rejected'])
            ])
            ->orderBy('reviewing_requests_count')
            ->firstOrFail();
    }

    private function recordEvent(
        StudentRequest $request,
        string $type,
        ?User $actor = null,
        ?string $to = null,
        ?string $comment = null
    ): void
    {
        RequestEvent::create([
            'request_id' => $request->id,
            'event_type' => $type,
            'from_state' => $request->getOriginal('state'),
            'to_state'   => $to,
            'actor_id'   => $actor?->id ?? auth()->id(),
            'comment'    => $comment,
        ]);
    }

    private function generateReference(): string
    {
        $year = now()->year;
        $seq  = StudentRequest::whereYear('created_at', $year)->count() + 1;
        return "REQ-{$year}-" . str_pad((string) $seq, 5, '0', STR_PAD_LEFT);
    }
}

6.13 SlaEscalationJob — تذكير وتصعيد تلقائي

// app/Jobs/Workflow/SlaEscalationJob.php
declare(strict_types=1);

namespace App\Jobs\Workflow;

use App\Models\StudentRequest;
use App\Models\User;
use App\Notifications\SlaWarningNotification;
use App\Notifications\SlaBreachedNotification;
use App\Services\Workflow\WorkflowService;
use Illuminate\Bus\Queueable;
use Illuminate\Contracts\Queue\ShouldQueue;
use Illuminate\Foundation\Bus\Dispatchable;
use Illuminate\Queue\InteractsWithQueue;
use Illuminate\Queue\SerializesModels;

final class SlaEscalationJob implements ShouldQueue
{
    use Dispatchable, InteractsWithQueue, Queueable, SerializesModels;

    public function handle(WorkflowService $workflow): void
    {
        // 1. طلبات قريبة من تجاوز SLA (خلال 24 ساعة)
        StudentRequest::query()
            ->whereBetween('sla_due_at', [now(), now()->addHours(24)])
            ->whereNotIn('state', ['completed', 'closed', 'rejected'])
            ->whereHas('currentReviewer')
            ->chunk(50, fn($batch) => $batch->each(
                fn(StudentRequest $r) => $r->currentReviewer->notify(
                    new SlaWarningNotification($r)
                )
            ));

        // 2. طلبات متجاوزة SLA — تصعيد للمستوى الأعلى
        StudentRequest::query()
            ->where('sla_due_at', '<', now())
            ->whereNotIn('state', ['completed', 'closed', 'rejected'])
            ->whereDoesntHave('events', fn($q) =>
                $q->where('event_type', 'escalated')
                  ->where('created_at', '>', now()->subDay())
            )
            ->chunk(50, function ($batch) {
                $batch->each(fn(StudentRequest $r) => $this->escalate($r));
            });
    }

    private function escalate(StudentRequest $request): void
    {
        // أعلَى مدير في نفس الفرع
        $manager = User::role('branch_manager')
            ->where('branch_id', $request->student->branch_id)
            ->first();

        if (! $manager) return;

        $manager->notify(new SlaBreachedNotification($request));
        $request->currentReviewer?->notify(new SlaBreachedNotification($request));

        $request->events()->create([
            'event_type' => 'escalated',
            'metadata'   => ['escalated_to' => $manager->id],
            'comment'    => 'تصعيد تلقائي لتجاوز SLA',
        ]);
    }
}

// app/Services/Letters/LetterGenerator.php
declare(strict_types=1);

namespace App\Services\Letters;

use App\Enums\LetterType;
use App\Models\Letter;
use App\Models\Student;
use Endroid\QrCode\Builder\Builder;
use Endroid\QrCode\Encoding\Encoding;
use Endroid\QrCode\Writer\PngWriter;
use Illuminate\Support\Facades\Storage;
use Illuminate\Support\Str;
use Spatie\Browsershot\Browsershot;

final class LetterGenerator
{
    public function __construct(
        private DigitalSigner $signer,
    ) {}

    public function generate(
        Student $student,
        LetterType $type,
        array $data = [],
        ?string $requestId = null,
    ): Letter
    {
        $letter = Letter::create([
            'reference'          => $this->generateReference(),
            'student_id'         => $student->id,
            'type'               => $type->value,
            'request_id'         => $requestId,
            'data'               => $data,
            'subject'            => $type->label(),
            'status'             => 'draft',
            'verification_token' => Str::random(48),
            'expires_at'         => now()->addMonths($type->validityMonths()),
        ]);

        $pdfPath = $this->renderPdf($letter);
        $signedPath = $this->signer->sign($pdfPath, $letter);

        $letter->addMedia($signedPath)
            ->usingFileName("{$letter->reference}.pdf")
            ->toMediaCollection('letter_pdfs');

        $letter->update([
            'status'     => 'signed',
            'signed_at'  => now(),
            'signed_by'  => auth()->id(),
        ]);

        \App\Models\LetterAudit::create([
            'letter_id'  => $letter->id,
            'action'     => 'generated',
            'actor_id'   => auth()->id(),
            'ip_address' => request()->ip(),
            'user_agent' => request()->userAgent(),
        ]);

        @unlink($pdfPath);
        @unlink($signedPath);

        return $letter->fresh();
    }

    private function renderPdf(Letter $letter): string
    {
        $qrCode = $this->generateQrCode($letter);
        $html = view("letters.{$letter->type->value}", [
            'letter'  => $letter,
            'subject' => $letter->subject,
            'qrCode'  => $qrCode,
        ])->render();

        $path = storage_path("app/temp/letter-{$letter->id}.pdf");

        Browsershot::html($html)
            ->setNodeBinary(config('browsershot.node_path'))
            ->setNpmBinary(config('browsershot.npm_path'))
            ->format('A4')
            ->showBackground()
            ->noSandbox()
            ->waitUntilNetworkIdle()
            ->emulateMedia('print')
            ->deviceScaleFactor(2)        // أوضح للطباعة
            ->save($path);

        return $path;
    }

    private function generateQrCode(Letter $letter): string
    {
        $url = route('public.letters.verify', $letter->verification_token);

        $result = Builder::create()
            ->writer(new PngWriter())
            ->data($url)
            ->encoding(new Encoding('UTF-8'))
            ->size(280)
            ->margin(6)
            ->build();

        return base64_encode($result->getString());
    }

    private function generateReference(): string
    {
        $year = now()->year;
        $seq = Letter::whereYear('created_at', $year)->count() + 1;
        return "LTR-{$year}-" . str_pad((string) $seq, 5, '0', STR_PAD_LEFT);
    }
}

7.8 DigitalSigner — التوقيع الرقمي بـTCPDF/FPDI

// app/Services/Letters/DigitalSigner.php
declare(strict_types=1);

namespace App\Services\Letters;

use App\Models\Letter;
use setasign\Fpdi\Tcpdf\Fpdi;

final class DigitalSigner
{
    public function sign(string $pdfPath, Letter $letter): string
    {
        $pdf = new Fpdi();
        $pageCount = $pdf->setSourceFile($pdfPath);

        for ($i = 1; $i <= $pageCount; $i++) {
            $tpl = $pdf->importPage($i);
            $size = $pdf->getTemplateSize($tpl);
            $pdf->AddPage($size['orientation'], [$size['width'], $size['height']]);
            $pdf->useTemplate($tpl);
        }

        // تطبيق توقيع رقمي بشهادة المعهد (PKCS#12)
        $certPath = storage_path('app/certs/iims-cert.p12');
        $certPwd  = config('letters.cert_password');

        $pdf->setSignature(
            "file://{$certPath}",
            "file://{$certPath}",
            $certPwd,
            '',
            2,
            [
                'Name'         => 'IIMS - Letters Authority',
                'Location'     => 'Riyadh, Saudi Arabia',
                'Reason'       => 'Official letter authentication',
                'ContactInfo'  => 'https://iims.local',
            ]
        );

        $signedPath = storage_path("app/temp/letter-{$letter->id}-signed.pdf");
        $pdf->Output($signedPath, 'F');

        return $signedPath;
    }
}

namespace App\Http\Controllers;

use App\Models\Letter;
use App\Models\LetterAudit;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\RateLimiter;
use Illuminate\View\View;

final class PublicLetterVerifyController extends Controller
{
    public function __invoke(Request $request, string $token): View
    {
        // منع الـbrute-force
        $key = 'verify:' . $request->ip();
        if (RateLimiter::tooManyAttempts($key, 30)) {
            abort(429, 'محاولات كثيرة. حاول لاحقاً.');
        }
        RateLimiter::hit($key, 60);

        $letter = Letter::with(['student.branch'])
            ->where('verification_token', $token)
            ->first();

        if (! $letter) {
            return view('letters.verify-not-found');
        }

        $valid = $letter->status === 'signed' || $letter->status === 'sent';
        $expired = $letter->expires_at?->isPast() ?? false;

        LetterAudit::create([
            'letter_id'  => $letter->id,
            'action'     => 'verified',
            'ip_address' => $request->ip(),
            'user_agent' => $request->userAgent(),
        ]);

        return view('letters.verify', [
            'letter'   => $letter,
            'valid'    => $valid && ! $expired,
            'expired'  => $expired,
        ]);
    }
}

// app/Listeners/AutoGenerateLetterOnRequestApproval.php
declare(strict_types=1);

namespace App\Listeners;

use App\Enums\LetterType;
use App\Enums\RequestType;
use App\Models\StudentRequest;
use App\Services\Letters\LetterGenerator;
use Spatie\ModelStates\Events\StateChanged;

final class AutoGenerateLetterOnRequestApproval
{
    public function __construct(private LetterGenerator $generator) {}

    public function handle(StateChanged $event): void
    {
        if (! $event->model instanceof StudentRequest) return;
        if ($event->newState->getValue() !== 'processing') return;

        $request = $event->model;

        $letterType = match($request->type) {
            RequestType::ENROLLMENT_LETTER    => LetterType::ENROLLMENT,
            RequestType::REPLACEMENT_CERT     => LetterType::REPLACEMENT_CERTIFICATE,
            RequestType::DEFER_SEMESTER       => LetterType::DEFERRAL_APPROVAL,
            RequestType::RESUME_STUDIES       => LetterType::RESUMPTION_APPROVAL,
            RequestType::TRANSFER_BRANCH      => LetterType::TRANSFER_APPROVAL,
            RequestType::PARTIAL_FEE_EXEMPT,
            RequestType::FULL_FEE_EXEMPT      => LetterType::FEE_EXEMPTION,
            RequestType::INSTALLMENT_PLAN     => LetterType::INSTALLMENT,
            default                           => null,
        };

        if (! $letterType) return;

        $this->generator->generate(
            student:   $request->student,
            type:      $letterType,
            data:      $request->data ?? [],
            requestId: $request->id,
        );
    }
}
