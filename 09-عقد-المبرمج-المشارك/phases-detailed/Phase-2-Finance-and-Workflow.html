Codex ليس مُراجِعاً لطيفاً. هو مستشار 100/100 صارم. مهمّته أن يحمي المشروع من قرارات هندسية متسرّعة. هذا البروتوكول مُلزِم في بداية كل أسبوع من المرحلة 2.

3.1 الخطوات السبع لاعتماد أسبوع

🤝 بروتوكول الاعتماد المُلزِم لكل أسبوع

  1. تحضير خطة الأسبوع — يُسلّم المبرمج المشارك ملفاً واحداً بصيغة Markdown (docs/phase-2/week-N-plan.md) يضم: الأهداف، المهام اليومية، Schema SQL، توقيع APIs، سيناريوهات الاختبار، وقائمة المخاطر.
  2. إرسال للـCodex عبر prompt قياسي: "راجع خطة المرحلة 2 الأسبوع N. قيّم على 100. ارفض إن وجدت أي ثغرة."
  3. تقييم Codex من 100 — يُعطي درجة + قائمة ملاحظات نقطية. درجة 92 = لا يكفي. مطلوب 100/100.
  4. التعديل والإعادة — يتم تعديل الخطة معالجة كل ملاحظة، ثم إعادة العرض. عادةً 2-3 جولات قبل 100/100.

  6. مراجعة يومية — في نهاية كل يوم عمل، يُرسل diff للـCodex مع وصف ما تمّ. أي ملاحظة Critical تُعالَج في اليوم التالي قبل أي شيء جديد.
  7. اعتماد الإنهاء — في نهاية الأسبوع، Codex يُقيّم النتيجة النهائية على 100. لا يُفتَح PR للـmain قبل 100/100 على المُسلَّم.

4.3 جداول قاعدة البيانات (SQL DDL كامل)

أربعة جداول جديدة. كلها تحت RLS، وكلها مفهرسة على أنماط الاستعلام المتوقّعة من محرّك الحجب.

-- =================================================================
-- 4.3.1 accounting_snapshot — لقطة الحالة المالية لكل طالب
-- المصدر الحقيقي يبقى المحاسبي. هذه نسخة سريعة للقراءة.
-- =================================================================
CREATE TABLE accounting_snapshot (
  student_id              UUID          PRIMARY KEY REFERENCES students(id) ON DELETE CASCADE,
  external_id             TEXT          NOT NULL,
  total_fees              NUMERIC(12,2) NOT NULL DEFAULT 0,
  total_paid              NUMERIC(12,2) NOT NULL DEFAULT 0,
  total_remaining         NUMERIC(12,2) GENERATED ALWAYS AS (total_fees - total_paid) STORED,
  overdue_amount          NUMERIC(12,2) NOT NULL DEFAULT 0,
  overdue_days            INTEGER       NOT NULL DEFAULT 0,
  last_payment_date       DATE,
  next_due_date           DATE,
  installment_status      TEXT          NOT NULL DEFAULT 'on_track'
    CHECK (installment_status IN ('on_track', 'late', 'critical', 'cleared')),
  raw_response            JSONB,                 -- آخر استجابة كاملة من المحاسبي
  last_synced_at          TIMESTAMPTZ   NOT NULL DEFAULT NOW(),
  source                  TEXT          NOT NULL DEFAULT 'accounting_api',
  source_version          VARCHAR(50)
);

CREATE INDEX idx_snapshot_overdue
  ON accounting_snapshot(overdue_days DESC)
  WHERE overdue_days > 0;

CREATE INDEX idx_snapshot_synced     ON accounting_snapshot(last_synced_at);
CREATE INDEX idx_snapshot_external   ON accounting_snapshot(external_id);
CREATE INDEX idx_snapshot_status     ON accounting_snapshot(installment_status);

-- RLS: الطالب يرى نفسه فقط. المالية ومدير الفرع يرون فرعهم. الأدمن يرى الكل.
ALTER TABLE accounting_snapshot ENABLE ROW LEVEL SECURITY;

CREATE POLICY snapshot_select_own ON accounting_snapshot
  FOR SELECT USING (
    student_id IN (SELECT id FROM students WHERE user_id = auth.uid())
    OR EXISTS (
      SELECT 1 FROM users u
      WHERE u.id = auth.uid()
        AND u.role IN ('admin', 'super_admin', 'finance')
    )
  );

-- =================================================================
-- 4.3.2 accounting_sync_log — سجل عمليات التزامن
-- =================================================================
CREATE TABLE accounting_sync_log (
  id                UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  sync_type         TEXT        NOT NULL CHECK (sync_type IN ('full', 'incremental', 'student_specific', 'webhook')),
  triggered_by      TEXT        NOT NULL CHECK (triggered_by IN ('cron', 'manual', 'on_demand', 'webhook')),
  status            TEXT        NOT NULL CHECK (status IN ('started', 'success', 'partial', 'failed')),
  records_processed INT         DEFAULT 0,
  records_succeeded INT         DEFAULT 0,
  records_failed    INT         DEFAULT 0,
  error_message     TEXT,
  request_metadata  JSONB,
  response_metadata JSONB,
  started_at        TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  completed_at      TIMESTAMPTZ,
  duration_ms       INT
);

CREATE INDEX idx_sync_log_status ON accounting_sync_log(status, started_at DESC);
CREATE INDEX idx_sync_log_date   ON accounting_sync_log(started_at DESC);

-- =================================================================
-- 4.3.3 integration_errors — أخطاء التكاملات (DLQ-like)
-- =================================================================
CREATE TABLE integration_errors (
  id              UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
  provider        VARCHAR(50) NOT NULL,
  endpoint        VARCHAR(255),
  http_method     VARCHAR(10),
  error_code      VARCHAR(50),
  error_message   TEXT,
  severity        VARCHAR(20) NOT NULL CHECK (severity IN ('low', 'medium', 'high', 'critical')),
  context         JSONB,
  retry_count     INT         DEFAULT 0,
  next_retry_at   TIMESTAMPTZ,
  resolved        BOOLEAN     NOT NULL DEFAULT FALSE,
  resolved_at     TIMESTAMPTZ,
  resolved_by     UUID        REFERENCES users(id),
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_int_err_unresolved
  ON integration_errors(provider, created_at DESC)
  WHERE resolved = FALSE;

-- =================================================================
-- 4.3.4 webhook_inbox — لتفادي تكرار معالجة webhook
-- =================================================================
CREATE TABLE webhook_inbox (
  idempotency_key TEXT        PRIMARY KEY,
  provider        VARCHAR(50) NOT NULL,
  event_type      VARCHAR(100),
  payload         JSONB       NOT NULL,
  processed_at    TIMESTAMPTZ,
  received_at     TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

لماذا GENERATED ALWAYS AS؟

حقل total_remaining محسوب من total_fees - total_paid. وضعه كـ Stored Generated Column يضمن الاتساق ويسمح بالفهرسة المباشرة عليه دون لمسات شيفرة. تحذير: إن تغيّر منطق الحساب لاحقاً، لا تنسَ Migration لإعادة توليد العمود.

export class ExternalAccountingAdapter implements IAccountingService {
  constructor(private readonly config: AccountingConfig) {}

  async getStudentBalance(studentId: string): Promise<StudentBalance> {
    const cacheKey = `acct:balance:${studentId}`;
    const cached = await redisCache.get<StudentBalance>(cacheKey);
    if (cached) return cached;

    const data = await callAccounting(async () => {
      return await httpClient.get<RawBalance>(
        `${this.config.baseUrl}/students/${studentId}/balance`,
        {
          headers: { 'X-API-Key': this.config.apiKey },
          timeoutMs: this.config.timeoutMs
        }
      );
    });

    const mapped = this.mapToBalance(data, studentId);
    await redisCache.set(cacheKey, mapped, this.config.cacheTtlSec);
    return mapped;
  }

  async getStudentPayments(studentId: string, since?: Date): Promise<Payment[]> {
    const qs = since ? `?since=${since.toISOString()}` : '';
    const data = await callAccounting(() =>
      httpClient.get<RawPayment[]>(
        `${this.config.baseUrl}/students/${studentId}/payments${qs}`,
        { headers: { 'X-API-Key': this.config.apiKey } }
      )
    );
    return data.map(p => this.mapToPayment(p));
  }

  async getBranchRevenue(branchId: string, period: { from: Date; to: Date }) {
    return await callAccounting(() =>
      httpClient.get<BranchRevenue>(
        `${this.config.baseUrl}/branches/${branchId}/revenue`,
        { params: { from: period.from.toISOString(), to: period.to.toISOString() } }
      )
    );
  }

  verifyWebhookSignature(signature: string, body: string): boolean {
    const expected = createHmac('sha256', this.config.webhookSecret)
      .update(body)
      .digest('hex');
    // timing-safe comparison
    if (signature.length !== expected.length) return false;
    let diff = 0;
    for (let i = 0; i < expected.length; i++) {
      diff |= signature.charCodeAt(i) ^ expected.charCodeAt(i);
    }
    return diff === 0;
  }

  async healthCheck() {
    const start = Date.now();
    try {
      await httpClient.get(`${this.config.baseUrl}/health`, { timeoutMs: 3000 });
      return { ok: true, latencyMs: Date.now() - start };
    } catch {
      return { ok: false, latencyMs: Date.now() - start };
    }
  }

  private mapToBalance(raw: RawBalance, studentId: string): StudentBalance {
    return {
      studentId,
      externalId: raw.external_id,
      totalFees: Number(raw.total_fees),
      paidAmount: Number(raw.total_paid),

      currency: 'SAR'

import { CircuitBreaker } from '@/lib/circuit-breaker';
import { retry } from '@/lib/retry';
import { db } from '@/server/db';
import { logger } from '@/server/logger';

export class IntegrationError extends Error {
  constructor(
    public readonly provider: string,
    public readonly originalError: unknown,
    public readonly severity: 'low' | 'medium' | 'high' | 'critical' = 'medium',
    public readonly context?: Record<string, unknown>
  ) {
    super(`[${provider}] ${originalError}`);
    this.name = 'IntegrationError';
  }
}

export const accountingCircuit = new CircuitBreaker({
  name: 'accounting',
  threshold: 5,           // 5 أخطاء متتالية يفتح الـcircuit
  timeout: 60_000,        // يبقى مفتوحاً 60 ثانية
  resetTimeout: 30_000,   // بعدها يحاول كل 30 ثانية
  onOpen: () => logger.error('Accounting circuit OPENED — fallback to snapshot'),
  onHalfOpen: () => logger.info('Accounting circuit HALF-OPEN'),
  onClose: () => logger.info('Accounting circuit CLOSED — back to normal'),
});

export async function callAccounting<T>(
  fn: () => Promise<T>,
  options?: { fallback?: () => Promise<T>; severity?: 'low' | 'medium' | 'high' | 'critical' }
): Promise<T> {
  try {
    return await accountingCircuit.execute(async () => {
      return await retry(fn, {
        attempts: 3,
        backoff: 'exponential',
        baseDelayMs: 500,
        maxDelayMs: 5000,
        retryOn: (err) => isTransient(err),
      });
    });
  } catch (err) {
    const severity = options?.severity ?? 'high';
    await db.integration_errors.insert({
      provider: 'accounting',
      endpoint: (fn as any).name ?? 'unknown',
      error_message: String(err),
      severity,
      context: { stack: (err as Error).stack },
    });

    if (options?.fallback) {
      logger.warn('Accounting call failed — using fallback');
      return options.fallback();
    }
    throw new IntegrationError('accounting', err, severity);
  }
}

function isTransient(err: unknown): boolean {
  if (!(err instanceof Error)) return false;
  const msg = err.message.toLowerCase();
  return msg.includes('timeout') ||
         msg.includes('econnreset') ||
         msg.includes('enotfound') ||
         msg.includes('502') || msg.includes('503') || msg.includes('504');
}

// src/app/api/webhooks/accounting/route.ts
import { NextResponse } from 'next/server';
import { createAccountingService } from '@/server/integrations/accounting';
import { db } from '@/server/db';
import { enqueueBlockingEvaluation } from '@/server/blocking/queue';
import { rateLimit } from '@/server/security/rate-limit';

export async function POST(req: Request) {
  // 1) Rate limit (per IP) — حماية أساسية
  const rl = await rateLimit(req, { key: 'webhook:accounting', limit: 60, windowSec: 60 });
  if (!rl.allowed) return NextResponse.json({ error: 'rate_limited' }, { status: 429 });

  // 2) Read raw body — لازم قبل JSON.parse للتحقق من التوقيع
  const rawBody = await req.text();
  const signature = req.headers.get('x-accounting-signature') ?? '';
  const idempotencyKey = req.headers.get('x-idempotency-key') ?? '';

  // 3) Verify HMAC signature (timing-safe)
  const adapter = createAccountingService();
  if (!adapter.verifyWebhookSignature(signature, rawBody)) {
    await db.integration_errors.insert({
      provider: 'accounting', endpoint: 'webhook',
      error_message: 'Invalid signature', severity: 'critical',
      context: { ip: req.headers.get('x-forwarded-for') }
    });
    return NextResponse.json({ error: 'invalid_signature' }, { status: 401 });
  }

  // 4) Idempotency: لا نُعالج نفس الحدث مرتين
  if (idempotencyKey) {
    const seen = await db.webhook_inbox.findUnique({ where: { idempotency_key: idempotencyKey } });
    if (seen?.processed_at) {
      return NextResponse.json({ status: 'already_processed' }, { status: 200 });
    }
  }

  // 5) Parse + store + dispatch
  const event = JSON.parse(rawBody);
  await db.webhook_inbox.upsert({
    where: { idempotency_key: idempotencyKey || `auto-${crypto.randomUUID()}` },
    create: { provider: 'accounting', event_type: event.type, payload: event },
    update: {}
  });

  switch (event.type) {
    case 'payment.received':
      await handlePaymentReceived(event);
      break;
    case 'invoice.issued':
      await handleInvoiceIssued(event);
      break;
    case 'installment.overdue':
      await enqueueBlockingEvaluation(event.student_external_id);
      break;
  }

  await db.webhook_inbox.update({
    where: { idempotency_key: idempotencyKey },
    data: { processed_at: new Date() }
  });

  return NextResponse.json({ status: 'ok' }, { status: 200 });
}

async function handlePaymentReceived(event: PaymentReceivedEvent) {
  // 1. Invalidate cache
  await redisCache.delete(`acct:balance:${event.student_id}`);
  // 2. Re-fetch balance and update snapshot
  const balance = await createAccountingService().getStudentBalance(event.student_id);
  await upsertSnapshot(balance);
  // 3. Re-evaluate blocking — قد تُلغى حجوب آلياً بعد الدفع
  await enqueueBlockingEvaluation(event.student_id);
}

5.5 المُقيِّم (Condition Evaluator)

// src/server/blocking/evaluator.ts
import type { BlockingCondition, EvaluationContext } from './types';

export function evaluateCondition(
  cond: BlockingCondition,
  ctx: EvaluationContext
): boolean {
  switch (cond.kind) {
    case 'overdue_days_gt':
      return ctx.balance.overdueDays > cond.value;
    case 'overdue_days_gte':
      return ctx.balance.overdueDays >= cond.value;
    case 'overdue_amount_gt':
      return ctx.balance.overdueAmount > cond.value;
    case 'student_status_in':
      return cond.values.includes(ctx.status);
    case 'absence_rate_gte':
      return ctx.absenceRate >= cond.value;
    case 'has_pending_promise':
      return ctx.hasPendingPromise === cond.value;
    case 'and':
      return cond.rules.every(r => evaluateCondition(r, ctx));
    case 'or':
      return cond.rules.some(r => evaluateCondition(r, ctx));
    case 'not':
      return !evaluateCondition(cond.rule, ctx);
  }
}

// src/server/blocking/engine.ts
import { db } from '@/server/db';
import { createAccountingService } from '@/server/integrations/accounting';
import { evaluateCondition } from './evaluator';
import { notifyStudent } from '@/server/notifications';
import type { BlockResult, ServiceKey } from './types';

export async function evaluateBlocking(studentId: string): Promise<BlockResult> {
  // 1) بناء سياق التقييم
  const [snapshot, student, absence, promise] = await Promise.all([
    createAccountingService().getStudentBalance(studentId),
    db.students.findUnique({ where: { id: studentId } }),
    getAverageAbsenceRate(studentId),
    hasPendingPaymentPromise(studentId),
  ]);

  if (!student) throw new Error(`Student ${studentId} not found`);

  const ctx: EvaluationContext = {
    studentId, status: student.status,
    balance: snapshot, absenceRate: absence,
    hasPendingPromise: promise, evaluatedAt: new Date()
  };

  // 2) جلب القواعد النشطة مرتّبة بالأولوية
  const rules = await db.blocking_rules.findMany({
    where: { is_active: true },
    orderBy: { priority: 'desc' }
  });

  // 3) جلب overrides النشطة لهذا الطالب
  const overrides = await db.rule_overrides.findMany({
    where: {
      student_id: studentId,
      is_active: true,
      OR: [{ expires_at: null }, { expires_at: { gt: new Date() } }]
    }
  });

  // 4) تطبيق كل قاعدة → جمع الخدمات المحجوبة والتحذيرات
  const blocked: BlockResult['blocked'] = [];
  const warnings: BlockResult['warnings'] = [];

  for (const rule of rules) {
    if (!evaluateCondition(rule.condition, ctx)) continue;

    // شطب الخدمات المُغطَّاة بـoverride لهذه القاعدة
    const override = overrides.find(o => o.rule_id === rule.id);
    const effectiveServices = override
      ? rule.blocked_services.filter(s => !override.services_overridden.includes(s))
      : rule.blocked_services;

    if (effectiveServices.length === 0) continue;

    if (rule.warning_only) {
      warnings.push({
        ruleId: rule.id, services: effectiveServices,
        message: rule.name_ar
      });
    } else {
      blocked.push({
        ruleId: rule.id, ruleName: rule.name_ar,
        services: effectiveServices,
        reason: rule.name_ar
      });
    }
  }

  // 5) Diff مع الحالة الحالية + تطبيق التغييرات
  await applyBlockingDiff(studentId, blocked, ctx);

  return {
    studentId, blocked, warnings,
    overridden: overrides.map(o => ({ overrideId: o.id, services: o.services_overridden }))
  };
}

// src/server/blocking/apply.ts
async function applyBlockingDiff(
  studentId: string,
  desired: BlockResult['blocked'],
  ctx: EvaluationContext
) {
  const existing = await db.service_blocks.findMany({
    where: { student_id: studentId, is_active: true, block_type: 'automatic' }
  });

  // (أ) قواعد كانت تطبّق ولم تعد → إلغاء الحجب
  for (const ex of existing) {
    if (!desired.some(d => d.ruleId === ex.rule_id)) {
      await db.service_blocks.update({
        where: { id: ex.id },
        data: {
          is_active: false, unblocked_at: new Date(),
          unblock_reason: 'rule_no_longer_applies',
        }
      });
      await audit('BLOCK_AUTO_CLEARED', { blockId: ex.id, studentId, ruleId: ex.rule_id });
      await notifyStudent(studentId, 'block_cleared', { ruleId: ex.rule_id });
    }
  }

  // (ب) قواعد جديدة → إضافة
  for (const d of desired) {
    const already = existing.find(e => e.rule_id === d.ruleId);
    if (already) {
      // تحقق إن كانت الخدمات تغيّرت
      if (!arrayEq(already.blocked_services, d.services)) {
        await db.service_blocks.update({
          where: { id: already.id },
          data: { blocked_services: d.services, metadata: { lastReevaluatedAt: ctx.evaluatedAt } }
        });
      }
      continue;
    }
    const newBlock = await db.service_blocks.create({
      data: {
        student_id: studentId, rule_id: d.ruleId,
        block_type: 'automatic', reason: d.reason,
        blocked_services: d.services,
        metadata: {
          snapshot_balance: ctx.balance,
          snapshot_status: ctx.status,
          evaluated_at: ctx.evaluatedAt
        }
      }
    });
    await audit('BLOCK_AUTO_CREATED', { blockId: newBlock.id, studentId, ruleId: d.ruleId });
    await notifyStudent(studentId, 'block_created', {
      ruleId: d.ruleId, services: d.services, reason: d.reason
    });
  }
}

// src/server/blocking/unblock.ts
export interface UnblockRequest {
  blockId: string;
  reason: string;                  // ≥ 20 حرفاً
  unblockUntil?: Date;             // مؤقت أم دائم
  servicesToUnblock?: ServiceKey[]; // قد يفك جزئياً
  attachments?: string[];          // روابط مستندات داعمة
}

export async function unblockManually(
  req: UnblockRequest,
  actor: User
): Promise<{ overrideId: string }> {
  // 1) فحص الصلاحية
  if (!actor.permissions.includes('service_block.unblock')) {
    throw new ForbiddenError('لا تملك صلاحية فك الحجب');
  }
  // 2) فحص السبب
  if (req.reason.trim().length < 20) {
    throw new ValidationError('السبب يجب أن يكون 20 حرفاً على الأقل');
  }

  const block = await db.service_blocks.findUnique({ where: { id: req.blockId } });
  if (!block || !block.is_active) {
    throw new ValidationError('الحجب غير نشط');
  }

  // 3) فحص حد الصلاحية المالية (للمتأخرات الكبيرة) — يحتاج تأكيد العميل
  const snapshot = block.metadata?.snapshot_balance as StudentBalance | undefined;
  if (
    actor.role === 'branch_manager' &&
    snapshot && snapshot.overdueAmount > 5000
  ) {
    throw new ForbiddenError('متأخر > 5000 — يحتاج اعتماد الإدارة العامة');
  }

  // 4) سجّل override
  const services = req.servicesToUnblock ?? block.blocked_services;
  const override = await db.rule_overrides.create({
    data: {
      student_id: block.student_id, rule_id: block.rule_id,
      services_overridden: services, reason: req.reason,
      expires_at: req.unblockUntil ?? null,
      created_by: actor.id, attachments: req.attachments ?? []
    }
  });

  // 5) إن كان فك حجب كامل لهذا الـblock → ضع is_active=false
  const isFullUnblock = services.length === block.blocked_services.length;
  if (isFullUnblock) {
    await db.service_blocks.update({
      where: { id: block.id },
      data: {
        is_active: false, unblocked_at: new Date(),
        unblocked_by: actor.id, unblock_reason: req.reason
      }
    });
  }

  await audit('BLOCK_MANUAL_UNBLOCK', {
    blockId: block.id, overrideId: override.id,
    studentId: block.student_id, actor: actor.id,
    services, reason: req.reason,
    until: req.unblockUntil
  });

  await notifyStudent(block.student_id, 'manual_unblock', {
    services, until: req.unblockUntil
  });

  return { overrideId: override.id };
}

export function isWorkingMoment(d: Date, cfg: SLAConfig = defaultSLAConfig): boolean {
  const day = d.getDay();
  if (!cfg.workingDays.includes(day)) return false;

  const dateOnly = d.toISOString().slice(0, 10);
  if (cfg.holidays.some(h => h.toISOString().startsWith(dateOnly))) return false;

  const h = d.getHours();
  return h >= cfg.businessHours.startHour && h < cfg.businessHours.endHour;
}

function nextWorkingMoment(d: Date, cfg: SLAConfig): Date {
  let next = new Date(d);
  while (!isWorkingMoment(next, cfg)) {
    next = addMinutes(next, 15);
    // fast-forward to next business hour if outside
    if (next.getHours() >= cfg.businessHours.endHour) {
      next = setHours(setMinutes(addDays(next, 1), 0), cfg.businessHours.startHour);
    }
  }
  return next;
}

export function calculateSLADeadline(
  createdAt: Date,
  slaHours: number,
  cfg: SLAConfig = defaultSLAConfig
): Date {
  let remainingMin = slaHours * 60;
  let cursor = nextWorkingMoment(createdAt, cfg);

  while (remainingMin > 0) {
    if (isWorkingMoment(cursor, cfg)) {
      cursor = addMinutes(cursor, 15);
      remainingMin -= 15;
    } else {
      cursor = nextWorkingMoment(cursor, cfg);
    }
  }
  return cursor;
}

import { z } from 'zod';
import { routingRules } from './routing';
import { calculateSLADeadline } from './sla';
import { canAccessService } from '@/server/blocking';

const CreateRequestSchema = z.object({
  type: z.enum(['intro_letter', 'training_letter', /* 14 types */]),
  studentId: z.string().uuid(),
  data: z.record(z.unknown()),
  attachments: z.array(z.string().url()).optional(),
  priority: z.enum(['low', 'normal', 'high', 'critical']).default('normal'),
});

export async function createRequest(
  input: z.infer<typeof CreateRequestSchema>,
  actor: User
): Promise<{ requestId: string; serial: string }> {
  const parsed = CreateRequestSchema.parse(input);
  const rule = routingRules[parsed.type];

  // 1. فحص الـService Blocking قبل الإنشاء
  if (rule.blockedByFinanceBlock) {
    const serviceKey = mapRequestTypeToService(parsed.type); // e.g. intro_letter → letter_issuance
    const check = await canAccessService(parsed.studentId, serviceKey, {
      logCheck: true, actor: { id: actor.id }
    });
    if (!check.allowed) {
      throw new ServiceBlockedError(check.reason ?? 'الخدمة محجوبة');
    }
  }

  // 2. تحقّق من الحقول المطلوبة
  const missing = rule.requiredFields.filter(f => !(f in parsed.data));
  if (missing.length > 0) {
    throw new ValidationError(`حقول ناقصة: ${missing.join(', ')}`);
  }

  // 3. حساب SLA Deadline
  const slaConfig = await db.sla_config.findUnique({ where: { request_type: parsed.type } });
  const hours = parsed.priority === 'critical' ? (slaConfig?.critical_hours ?? slaConfig!.hours) : slaConfig!.hours;
  const deadline = calculateSLADeadline(new Date(), hours);

  // 4. إنشاء الطلب
  const serial = await generateRequestSerial();
  const request = await db.requests.create({
    data: {
      serial, type: parsed.type, status: 'new',
      priority: parsed.priority,
      student_id: parsed.studentId,
      branch_id: actor.branchId,
      created_by: actor.id,
      current_department: rule.initialDepartment,
      data: parsed.data,
      attachments: parsed.attachments ?? [],
      sla_deadline: deadline,
    },
  });

  // 5. تعيين تلقائي (Round Robin) لموظف في القسم
  const assignee = await pickAssignee(rule.initialAssigneeRole, actor.branchId);
  if (assignee) {
    await transitionRequest(request.id, { type: 'ASSIGN', assigneeId: assignee.id }, 'system');
  }

  // 6. سجل الحدث + إشعار الطالب
  await db.request_events.create({
    data: { request_id: request.id, event_type: 'CREATED', to_status: 'new', actor_id: actor.id }
  });
  await notifyStudent(parsed.studentId, 'request_created', { serial });

  return { requestId: request.id, serial };
}

// src/server/workflow/transition.ts
import { createActor } from 'xstate';
import { requestMachine } from './request-machine';

export async function transitionRequest(
  requestId: string,
  event: RequestEvent,
  actorId: string | 'system'
): Promise<{ newStatus: RequestStatus }> {
  const r = await db.requests.findUnique({ where: { id: requestId } });
  if (!r) throw new NotFoundError();

  // إعادة بناء سياق الـmachine من DB
  const ctx = await buildContext(r);

  // إنشاء actor مؤقت يبدأ من الحالة الحالية
  const actor = createActor(requestMachine, { input: ctx, snapshot: { value: r.status } as any });
  actor.start();

  const oldStatus = actor.getSnapshot().value as RequestStatus;

  actor.send(event);
  const newStatus = actor.getSnapshot().value as RequestStatus;

  if (oldStatus === newStatus) {
    throw new InvalidTransitionError(`${event.type} not allowed from ${oldStatus}`);
  }

  // كتابة التغيير في DB (transaction)
  await db.$transaction(async (tx) => {
    await tx.requests.update({
      where: { id: requestId },
      data: {
        status: newStatus,
        sla_paused_at: newStatus === 'awaiting_student' ? new Date() : null,
        assigned_to: event.type === 'ASSIGN' ? event.assigneeId : r.assigned_to,
        closed_at: newStatus === 'resolved_closed' ? new Date() : null,
        resolution: event.type === 'APPROVE' ? 'completed' :
                    event.type === 'REJECT'  ? 'rejected'  : null,
        resolution_note: 'reason' in event ? event.reason : null,
      },
    });
    await tx.request_events.create({
      data: {
        request_id: requestId, event_type: event.type,
        from_status: oldStatus, to_status: newStatus,
        actor_id: actorId === 'system' ? null : actorId,
        payload: event as any,
      }
    });
  });

  // أحداث ما بعد الانتقال
  if (event.type === 'APPROVE') await handleApproval(r, event);
  await notifyOnTransition(r, oldStatus, newStatus);

  return { newStatus };
}

Handlebars.registerHelper('formatDate', (date: Date | string, fmt?: string) => {
  const d = typeof date === 'string' ? new Date(date) : date;
  return format(d, fmt ?? 'dd MMMM yyyy', { locale: ar });
});

Handlebars.registerHelper('hijriDate', (date: Date | string) => {
  const d = typeof date === 'string' ? new Date(date) : date;
  const hijri = toHijri(d.getFullYear(), d.getMonth() + 1, d.getDate());
  return `${hijri.hy}/${hijri.hm}/${hijri.hd}`;
});

Handlebars.registerHelper('currency', (n: number) =>
  new Intl.NumberFormat('ar-SA', { style: 'currency', currency: 'SAR' }).format(n));

Handlebars.registerHelper('eq', (a, b) => a === b);

export function compileTemplate(htmlSource: string, cssSource: string): CompiledTemplate {
  const compiled = Handlebars.compile(htmlSource, { strict: true });
  return {
    render(data: Record<string, unknown>) {
      const body = compiled(data);
      // inject CSS inside <style> إن لم يكن مُضمَّناً مسبقاً
      return body.includes('</style>') ? body : body.replace('</head>', `<style>${cssSource}</style></head>`);
    }
  };
}

export interface IssueLetterInput {
  templateCode: string;
  studentId: string;
  requestId?: string;
  variables: Record<string, unknown>;
  optionalSections?: string[];
  issuedBy: string;
  approvedBy?: string;
}

export async function issueLetter(input: IssueLetterInput): Promise<Letter> {
  // 1) جلب القالب + الطالب + الفرع
  const [tpl, student, institute] = await Promise.all([
    db.letter_templates.findUnique({ where: { code: input.templateCode, is_active: true } }),
    db.students.findUniqueWithBranch(input.studentId),
    db.institute_settings.getSingleton()
  ]);
  if (!tpl) throw new NotFoundError(`Template ${input.templateCode}`);

  // 2) توليد Serial فريد
  const serial = await db.$queryRaw<{ s: string }>(`SELECT generate_letter_serial() as s`);
  const issuedAt = new Date();

  // 3) توليد QR (Pre-image: URL تتضمن serial + short-hash)
  const verifyUrl = `${process.env.PUBLIC_URL}/verify-letter/${serial.s}`;
  const qrDataUrl = await QRCode.toDataURL(verifyUrl, {
    errorCorrectionLevel: 'M', width: 300, margin: 1
  });

  // 4) تجميع المتغيّرات الكاملة
  const renderData = {
    serial: serial.s,
    issue_date_gregorian: issuedAt.toLocaleDateString('ar-SA'),
    issue_date_hijri: Handlebars.helpers.hijriDate(issuedAt),
    student: {
      full_name: student.full_name,
      id_number: student.id_number,
      branch_name: student.branch.name_ar,
      program_name: student.program.name_ar,
      enrollment_date: student.enrolled_at,
      status_label: getStatusLabel(student.status),
    },
    institute,
    issuer: await getIssuerInfo(input.issuedBy),
    qr_url: qrDataUrl,
    verify_url: verifyUrl,
    stamp_url: tpl.requires_stamp ? institute.stamp_url : null,
    letterhead: { logo_url: institute.logo_url },
    ...input.variables,
  };

  // 5) Render HTML
  const compiled = compileTemplate(tpl.html_template, tpl.css_style);
  const html = compiled.render(renderData);

  // 6) Puppeteer → PDF
  const browser = await getBrowser();
  const page = await browser.newPage();
  try {
    await page.setContent(html, { waitUntil: 'networkidle0', timeout: 15000 });
    await page.evaluateHandle('document.fonts.ready');
    await page.emulateMediaType('print');
    const pdfBuf = await page.pdf({
      format: 'A4',
      printBackground: true,
      preferCSSPageSize: true,
      margin: { top: '25mm', right: '20mm', bottom: '25mm', left: '20mm' }
    });

    // 7) Integrity hash (PDF + serial + issuedAt)
    const hash = createHash('sha256')
      .update(pdfBuf).update(serial.s).update(issuedAt.toISOString())
      .digest('hex');

    // 8) رفع إلى Supabase Storage
    const path = `letters/${student.branch_id}/${issuedAt.getFullYear()}/${issuedAt.getMonth() + 1}/${serial.s}.pdf`;
    await supabaseAdmin.storage.from('letters').upload(path, pdfBuf, {
      contentType: 'application/pdf', cacheControl: '3600', upsert: false
    });

    // 9) إنشاء سجل في DB
    const retentionUntil = new Date();
    retentionUntil.setFullYear(retentionUntil.getFullYear() + tpl.retention_years);

    const letter = await db.letters.create({
      data: {
        serial: serial.s, template_id: tpl.id, template_code: tpl.code,
        template_version: tpl.version, request_id: input.requestId ?? null,
        student_id: input.studentId, branch_id: student.branch_id,
        variables: input.variables, optional_sections: input.optionalSections ?? [],
        pdf_path: path, pdf_size_bytes: pdfBuf.length,
        integrity_hash: hash, qr_data: verifyUrl,
        issued_by: input.issuedBy, approved_by: input.approvedBy ?? null,
        status: 'issued', issued_at: issuedAt, retention_until: retentionUntil
      }
    });

    await audit('LETTER_ISSUED', { letterId: letter.id, serial: serial.s, studentId: input.studentId });
    await notifyStudent(input.studentId, 'letter_ready', { serial: serial.s, templateCode: tpl.code });

    return letter;
  } finally {
    await page.close();
  }
}

function maskName(name: string): string {
  const parts = name.split(' ');
  if (parts.length < 3) return name;
  parts[1] = parts[1][0] + '*****';
  return parts.join(' ');
}

  temp_suspension:  'SUSPENSION_LETTER',
  resume_studies:   'RESUME_STUDIES',
  // ... أنواع أخرى
};

export async function generateLetterFromRequest(req: Request): Promise<Letter> {
  const templateCode = requestTypeToTemplateCode[req.type];
  if (!templateCode) {
    throw new Error(`No template mapped for ${req.type}`);
  }

  // نقل البيانات من request.data إلى متغيرات القالب
  const variables = mapRequestDataToTemplateVars(req.type, req.data);

  return await issueLetter({
    templateCode,
    studentId: req.student_id,
    requestId: req.id,
    variables,
    issuedBy: req.assigned_to ?? req.created_by,
    approvedBy: req.assigned_to ?? undefined
  });
}
